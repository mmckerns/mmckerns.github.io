1#!/usr/bin/env python
2#
3# Problem definition:
4# Example in reference documentation for scipy.optimize.slsqp.
5# http://docs.scipy.org/doc/scipy-0.10.0/reference/tutorial/optimize.html
6#
7# Author: Mike McKerns (mmckerns @caltech and @uqfoundation)
8# Copyright (c) 1997-2016 California Institute of Technology.
9# License: 3-clause BSD.  The full license text is available at:
10#  - http://mmckerns.github.io/project/mystic/browser/mystic/LICENSE
11"""
12    Maximize: f = 2*x[0]*x[1] + 2*x[0] - x[0]**2 - 2*x[1]**2
13
14    Subject to: -2*x[0] + 2*x[1] <= -2
15                 2*x[0] - 4*x[1] <= 0
16                   x[0]**3 -x[1] == 0
17
18    where: 0 <= x[0] <= inf
19           1 <= x[1] <= inf
20"""
21
22import numpy as np
23import mystic.symbolic as ms
24import mystic.solvers as my
25import mystic.math as mm
26
27# generate constraints and penalty for a nonlinear system of equations
28ieqn = '''
29   -2*x0 + 2*x1 <= -2
30    2*x0 - 4*x1 <= 0
31'''
32eqn = '''
33     x0**3 - x1 == 0
34'''
35cons = ms.generate_constraint(ms.generate_solvers(ms.simplify(eqn,target='x1')))
36pens = ms.generate_penalty(ms.generate_conditions(ieqn), k=1e3)
37bounds = [(0., None), (1., None)]
38
39# get the objective
40def objective(x, sign=1):
41  x = np.asarray(x)
42  return sign * (2*x[0]*x[1] + 2*x[0] - x[0]**2 - 2*x[1]**2)
43
44x0 = np.random.rand(2)
45
46# compare against the exact minimum
47xs = np.array([2., 1.])
48ys = objective(xs, -1)
49
50
51sol = my.fmin_powell(objective, x0, constraint=cons, penalty=pens, disp=False,
52                     bounds=bounds, gtol=3, ftol=1e-6, full_output=True,
53                     args=(-1,))
54
55assert mm.almostEqual(sol[0], xs, tol=1e-2)
56assert mm.almostEqual(sol[1], ys, tol=1e-2)
57
58
59sol = my.diffev(objective, bounds, constraint=cons, penalty=pens, disp=False,
60                bounds=bounds, npop=15, gtol=100, ftol=1e-6, full_output=True,
61                args=(-1,))
62
63assert mm.almostEqual(sol[0], xs, tol=1e-2)
64assert mm.almostEqual(sol[1], ys, tol=1e-2)
65
66
