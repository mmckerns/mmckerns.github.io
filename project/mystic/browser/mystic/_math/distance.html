18def Lnorm(weights, p=1):
19  "calculate L-p norm of weights"
20  # weights is a numpy array
21  # p is an int
22  from numpy import asarray, seterr, inf
23  weights = asarray(weights).flatten()
24  if not p:
25    w = float(len(weights[weights != 0.0])) # total number of nonzero elements
26  elif p == inf:
27    w = float(max(abs(weights)))
28  else:
29    orig = seterr(over='raise', invalid='raise')
30    try:
31      w = float(sum(abs(weights**p)))**(1./p)
32    except FloatingPointError: # use the infinity norm
33      w = float(max(abs(weights)))
34    seterr(**orig)
35  return w

66def manhattan_distance(x, xp=None, **kwds):
67  """1-D manhattan distance between points
68
69  d[ij] = | x[i] - x'[j] |
70
71Notes:
72  manhattan matrix where x'==x will be symmetric with zeros on the diagonal
73  use dmin=2 for forced upconversion of 1-D arrays
74"""
75  # dmin: force upconvert to x,x' to dimension >= dmin
76  dmin = kwds['dmin'] if 'dmin' in kwds else 0 # default dimension
77  from numpy import abs, asarray, newaxis as nwxs, zeros_like
78  from __builtin__ import max
79  # cast as arrays of the same dimension
80  x = asarray(x)
81  xp = zeros_like(x) if xp is None else asarray(xp)

83  while len(x.shape) < xsize: x = x[nwxs]
84  while len(xp.shape) < xsize: xp = xp[nwxs]

87    x = x.T[:,:,nwxs]; xp = xp.T[:,nwxs]

89    x = x[:,nwxs]#     x.T[:,nwxs];  xp = xp.T

176def minkowski(x,xp=None, up=True, dmin=0, p=3, axis=None):
177  """p-norm distance between points in euclidean space
178
179  d(p) = sum( |x[0] - x[0]'|^p, |x[1] - x[1]'|^p, ..., |x[n] - x[n]'|^p )^(1/p)
180
181Input:
182  x    = array of points, x
183  xp   = array pf points, x'
184
185Additional Input:
186  up   = True if upconvert x with x[:,newaxis]
187  dmin = upconvert to x,x' to dimension >= dmin
188  axis = if not None, reduce across the selected axis
189  p    = value of p for the p-norm (default = 3)
190
191Notes:
192  for standard array behavior, use up=False
193  for element-wise across all elements, use up=True
194"""
195  from numpy import seterr, inf
196  if p == inf: return chebyshev(x,xp,up=up,dmin=dmin,axis=axis)
197  d = euclidean_distance(x,xp,up=up,dmin=dmin)
198  orig = seterr(over='raise', invalid='raise')
199  try:
200      d = (d**p).sum(axis=axis)**(1./p)
201  except FloatingPointError: # use the infinity norm
202      d = d.max(axis=axis).astype(float)
203  seterr(**orig)
204  return d

287def lipschitz_distance(L, points1, points2, **kwds):
288  """calculate the lipschitz distance between two sets of datapoints
289
290Inputs:
291 L = list of lipschitz constants
292 points1 = dataset or list of 'datapoint' or 'cone.vertex' objects
293 points2 = dataset or list of 'datapoint' or 'cone.vertex' objects
294
295Additional Inputs:
296 tol = maximum acceptable deviation from shortness
297 cutoff = zero out distances less than cutoff; typically: tol, 0.0, or None
298
299Returns:
300 list of lipschitz distances
301
302Notes:
303 Each point x,y can be thought to have an associated double-cone with slope
304 equal to the lipschitz constant. Shortness with respect to another point is
305 defined by the first point not being inside the cone of the second. We can
306 allow for some error in shortness, a short tolerance 'tol', for which the
307 point x,y is some acceptable y-distance inside the cone. While very tightly
308 related, cutoff and tol play distinct roles; tol is subtracted from
309 calculation of the lipschitz_distance, while cutoff zeros out the value
310 of any element less than the cutoff.
311"""
312  #FIXME: merge with lipschitz cone (distance/contains)
313  x,y   = _get_xy(points1)
314  xp,yp = _get_xy(points2)
315
316  # get tolerance in y
317  tol = kwds.pop('tol', 0.0)
318  cutoff = tol  # default is to zero out distances less than tolerance
319  if 'cutoff' in kwds: cutoff = kwds.pop('cutoff')
320  if cutoff is True: cutoff = tol
321  elif cutoff is False: cutoff = None
322
323  # calculate the distance matrix
324  md = manhattan_distance(y,yp) - max(0.0, tol)
325  lm = lipschitz_metric(L,x,xp)
326  d = md - lm
327  # zero-out all distances less than tolerated
328  return infeasibility(d, cutoff)

331def graphical_distance(model, points, **kwds):
332  """find the radius(x') that minimizes the graph between reality, y = G(x),
333and an approximating function, y' = F(x')
334
335Inputs:
336  model = the model function, y' = F(x'), that approximates reality, y = G(x)
337  points = object of type 'datapoint' to validate against; defines y = G(x)
338
339Additional Inputs:
340  ytol = maximum acceptable difference |y - F(x')|; a single value
341  xtol = maximum acceptable difference |x - x'|; an iterable or single value
342  cutoff = zero out distances less than cutoff; typically: ytol, 0.0, or None
343  hausdorff = norm; where if given, ytol = |y - F(x')| + |x - x'|/norm
344
345Returns:
346  radius = minimum distance from x,G(x) to x',F(x') for each x
347
348Notes:
349  xtol defines the n-dimensional base of a pilar of height ytol, centered at
