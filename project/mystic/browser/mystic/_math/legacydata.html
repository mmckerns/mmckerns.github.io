13def _fails(filter, data=None):
14  """filter a data array for failures using a boolean filter
15
16  Inputs:
17    filter -- a list of booleans (same size as data)
18    data -- a list of data points
19
20  Note:
21    if data=None, return the pairs of indicies where data fails
22  """
23  from numpy import where, asarray
24  failures = asarray(where(filter == False)).T
25  # if data not given, then return indicies for failure
26  if not data: return failures
27  # otherwise, return the list of 'bad' points

31class lipschitzcone(list):
32  """ Lipschitz double cone around a data point, with vertex and slope
33
34 queries:
35  vertex -- coordinates of lipschitz cone vertex
36  slopes -- lipschitz slopes for the cone (should be same dimension as 'vertex')
37
38 methods:
39  contains -- return True if a given point is within the cone
40  distance -- sum of lipschitz-weighted distance between a point and the vertex
41"""
42
43  def __init__(self, datapoint, slopes=None):
44    self.vertex = datapoint
45    if not slopes: slopes = [NULLSLOPE]*len(self.vertex.position)
46    self.slopes = slopes
47    return
48
49  def __repr__(self):
50    return "cone(%s @%s)" % (self.slopes, self.vertex.position)
51
52  def distance(self, point):
53    """ sum of lipschitz-weighted distance between a point and the vertex """
54    x = point.position
55    Z = self.vertex.position
56    L = self.slopes
57    return sum([L[i]*abs(x[i] - Z[i]) for i in range(len(x))])
58
59  def contains(self, point):
60    """ return True if a given point is within the cone """
61    L = self.slopes
62    if any([i == NULLSLOPE for i in L]): return True
63    x = point.position
64    Z = self.vertex.position
65    if all([x[i] == Z[i] for i in range(len(x))]): return True
66    y = point.value
67    G = self.vertex.value
68    return abs(y - G) <= self.distance(point)

301  def short(self, data=None, L=None, blamelist=False, pairs=True, \
302                                     all=False, raw=False, **kwds):
303    """check for shortness with respect to given data (or self)
304
305Inputs:
306    data -- a collection of data points (if None, use the dataset itself)
307    L -- the lipschitz constant, if different from that of the dataset itself
308    blamelist -- if True, report which points are infeasible
309    pairs -- if True, report indicies of infeasible points
310    all -- if True, report results for each point (opposed to all points)
311    raw -- if True, report numerical results (opposed to boolean results)
312
313Additional Inputs:
314    tol -- maximum acceptable deviation from shortness
315    cutoff -- zero out distances less than cutoff; typically: tol, 0.0, or None
316
317Notes:
318    Each point x,y can be thought to have an associated double-cone with slope
319    equal to the lipschitz constant. Shortness with respect to another point is
320    defined by the first point not being inside the cone of the second. We can
321    allow for some error in shortness, a short tolerance 'tol', for which the
322    point x,y is some acceptable y-distance inside the cone. While very tightly
323    related, cutoff and tol play distinct roles; tol is subtracted from
324    calculation of the lipschitz_distance, while cutoff zeros out the value
325    of any element less than the cutoff.
326"""
327    tol = kwds['tol'] if 'tol' in kwds else 0.0 # get tolerance in y
328    # default is to zero out distances less than tolerance
329    cutoff = kwds['cutoff'] if 'cutoff' in kwds else tol
330    if cutoff is True: cutoff = tol
331    elif cutoff is False: cutoff = None
332
333    if L is None: L = self.lipschitz
334    if data is None: data = self
335    from mystic.math.distance import lipschitz_distance, is_feasible
336    # calculate the shortness
337    Rv = lipschitz_distance(L, self, data, **kwds)
338    ld = is_feasible(Rv, cutoff)
339    if raw:
340      x = Rv
341    else:
342      x = ld
343    if not blamelist: return ld.all() if not all else x
344    if pairs: return _fails(ld)
345    # else lookup failures
346    return _fails(ld, data)

348  def valid(self, model, blamelist=False, pairs=True, \
349                         all=False, raw=False, **kwds):
350    """check for validity with respect to given model
351
352Inputs:
353    model -- the model function, y' = F(x')
354    blamelist -- if True, report which points are infeasible
355    pairs -- if True, report indicies of infeasible points
356    all -- if True, report results for each point (opposed to all points)
357    raw -- if True, report numerical results (opposed to boolean results)
358
359Additional Inputs:
360    ytol -- maximum acceptable difference |y - F(x')|; a single value
361    xtol -- maximum acceptable difference |x - x'|; an iterable or single value
362    cutoff -- zero out distances less than cutoff; typically: ytol, 0.0, or None
363    hausdorff -- norm; where if given, ytol = |y - F(x')| + |x - x'|/norm
364
365Notes:
366    xtol defines the n-dimensional base of a pilar of height ytol, centered at
367    each point. The region inside the pilar defines the space where a "valid"
368    model must intersect. If xtol is not specified, then the base of the pilar
369    will be a dirac at x' = x. This function performs an optimization for each
370    x to find an appropriate x'. While cutoff and ytol are very tightly related,
371    they play a distinct role; ytol is used to set the optimization termination
372    for an acceptable |y - F(x')|, while cutoff is applied post-optimization.
373    If we are using the hausdorff norm, then ytol will set the optimization
374    termination for an acceptable |y - F(x')| + |x - x'|/norm, where the x
375    values are normalized by norm = hausdorff.
376"""
377    ytol = kwds['ytol'] if 'ytol' in kwds else 0.0 # get tolerance in y
378    # default is to zero out distances less than tolerance
379    cutoff = kwds['cutoff'] if 'cutoff' in kwds else ytol
380    if cutoff is True: cutoff = ytol
381    elif cutoff is False: cutoff = None
382
383    from mystic.math.distance import graphical_distance, is_feasible
384    # calculate the model validity
385    Rv = graphical_distance(model, self, **kwds)
386    ld = is_feasible(Rv, cutoff)
387    if raw:
388      x = Rv
389    else:
390      x = ld
391    if not blamelist: return ld.all() if not all else x
392    if pairs: return _fails(ld)
393    # else lookup failures
394    return _fails(ld, self)

586def load_dataset(filename, filter=None):
587  """ read dataset from selected file
588
589  filename -- string name of dataset file
590  filter -- tuple of points to select ('False' to ignore filter stored in file)
591"""
592  from os.path import split, splitext
593  name = splitext(split(filename)[-1])[0]  # default name is filename
594  lipschitz = None
595  f = open(filename,"r")
596  file = f.read()
597  f.close()
598  contents = file.split("\n")
599  # parse file contents to get (i,id), cost, and parameters
600  pt = []; _filter = None
601  for line in contents[:-1]:
602    if line.startswith("# data name"):
603      name = eval(line.split(":")[-1])
604    elif line.startswith("# lipschitz"):
605      lipschitz = list( eval(line.split(":")[-1]) )
606    elif line.startswith("# filter") and filter != False:
607      _filter = eval(line.split(":")[-1])
608    elif line.startswith("#"):
609      pass
610    else:
611      data = line.split("   ")
612      sid = eval(data[0])
613      value = eval(data[1])
614      coords = list(eval(data[-1]))
615      pt.append(datapoint(list(coords), value, id=sid))
616
617  # apply filter(s)
618  from numpy import asarray
619  if _filter is not None and filter != False:
620    _filter = asarray(_filter)
621    _filter = _filter[_filter < len(pt)]
622    pt = [pt[i] for i in _filter]
623  if filter is not None and filter != False:
