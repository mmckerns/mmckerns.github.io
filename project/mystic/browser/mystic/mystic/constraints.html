83def with_mean(target):
84    """bind a mean constraint to a given constraints function.
85
86Inputs:
87    target -- the target mean
88
89A constraints function takes an iterable x as input, returning a modified x.
90This function is an "outer" coupling of "impose_mean" onto another constraints
91function c(x), such that:  x' = impose_mean(target, c(x)).
92
93    For example:
94    >>> @with_mean(5.0)
95    ... def constraint(x):
96    ...   x[-1] = x[0]
97    ...   return x
98    ...
99    >>> x = constraint([1,2,3,4])
100    >>> print x
101    [4.25, 5.25, 6.25, 4.25]
102    >>> mean(x)
103    5.0
104    """
105    def decorate(constraints):
106        def factory(x, *args, **kwds):
107            # apply decorated constraints function
108            x = constraints(x, *args, **kwds)
109            # constrain x such that mean(x) == target
110            if not almostEqual(mean(x), target):
111                x = impose_mean(target, x)#, weights=weights)
112            return x
113        return factory
114    return decorate

214    mass -- the target sum of normalized weights

240def issolution(constraints, guess, tol=1e-3):
241    """Returns whether the guess is a solution to the constraints
242
243Input:
244    constraints -- a constraints solver function or a penalty function
245    guess -- list of parameter values proposed to solve the constraints
246    tol -- residual error magnitude for which constraints are considered solved
247
248    For example:
249    >>> @normalized()
250    ... def constraint(x):
251    ...   return x
252    ...
253    >>> constraint([.5,.5])
254    [0.5, 0.5]
255    >>> issolution(constraint, [.5,.5])
256    True
257    >>>
258    >>> from mystic.penalty import quadratic_inequality
259    >>> @quadratic_inequality(lambda x: x[0] - x[1] + 10)
260    ... def penalty(x):
261    ...   return 0.0
262    ...
263    >>> penalty([-10,.5])
264    0.0
265    >>> issolution(penalty, [-10,.5])
266    True
267    """
268    if hasattr(constraints, 'error'):
269        error = constraints.error(guess)
270    else: # is a constraints solver
271        try:
272            constrained = guess.copy()
273        except:
274            from copy import deepcopy
275            constrained = deepcopy(guess)
276        error = 0.0
277        constrained = constraints(constrained)
278        for i in range(len(guess)):
279            error += (constrained[i] - guess[i])**2  #XXX: better rnorm ?
280        error = error**0.5
281
282    return error <= tol

288def solve(constraints, guess=None, nvars=None, solver=None, \
289          lower_bounds=None, upper_bounds=None, termination=None):
290    """Use optimization to find a solution to a set of constraints.
291
292Inputs:
293    constraints -- a constraints solver function or a penalty function
294
295Additional Inputs:
296    guess -- list of parameter values proposed to solve the constraints.
297    lower_bounds -- list of lower bounds on solution values.
298    upper_bounds -- list of upper bounds on solution values.
299    nvars -- number of parameter values.
300    solver -- the mystic solver to use in the optimization
301    termination -- the mystic termination to use in the optimization
302
303NOTE: The resulting constraints will likely be more expensive to evaluate
304    and less accurate than writing the constraints solver from scratch.
305
306NOTE: The ensemble solvers are available, using the default NestedSolver,
307    where the keyword 'guess' can be used to set the number of solvers.
308
309NOTE: The default solver is 'diffev', with npop=min(40, ndim*5). The default
310    termination is ChangeOverGeneration(), and the default guess is randomly
311    selected points between the upper and lower bounds.
312    """
313    npts = 8
314    if type(guess) is int: npts, guess = guess, None
315
316    ndim = 1 #XXX: better, increase in while loop catching IndexError ?
317    if nvars is not None: ndim = nvars
318    elif guess is not None: ndim = len(guess)
319    elif lower_bounds is not None: ndim = len(lower_bounds)
320    elif upper_bounds is not None: ndim = len(upper_bounds)
321
322    def cost(x): return 1.
323
324    #XXX: don't allow solver string as a short-cut? #FIXME: add ensemble solvers
325    ensemble = False
326    if solver is None or solver == 'diffev':
327        from mystic.solvers import DifferentialEvolutionSolver as TheSolver
328        solver = TheSolver(ndim, min(40, ndim*5))
329    elif solver == 'diffev2':
330        from mystic.solvers import DifferentialEvolutionSolver2 as TheSolver
331        solver = TheSolver(ndim, min(40, ndim*5))
332    elif solver == 'fmin_powell': #XXX: better as the default? (it's not random)
333        from mystic.solvers import PowellDirectionalSolver as TheSolver
334        solver = TheSolver(ndim)
335    elif solver == 'fmin':
336        from mystic.solvers import NelderMeadSimplexSolver as TheSolver
337        solver = TheSolver(ndim)
338    elif solver == 'buckshot':
339        from mystic.solvers import BuckshotSolver as TheSolver
340        solver = TheSolver(ndim, max(8, npts)) #XXX: needs better default?
341        ensemble = True
342    elif solver == 'lattice':
343        from mystic.solvers import LatticeSolver as TheSolver
344        solver = TheSolver(ndim, max(8, npts)) #XXX: needs better default?
345        ensemble = True
346   
347    if termination is None:
348        from mystic.termination import ChangeOverGeneration as COG
349        termination = COG()
350    if not ensemble:
351        if guess is not None:
352            solver.SetInitialPoints(guess) #XXX: nice if 'diffev' had methods
353        else:
354            solver.SetRandomInitialPoints(lower_bounds, upper_bounds)
355    if lower_bounds or upper_bounds:
356        solver.SetStrictRanges(lower_bounds, upper_bounds)
357    if hasattr(constraints, 'iter') and hasattr(constraints, 'error'):
358        solver.SetPenalty(constraints) #i.e. is a penalty function
359    else: # is a constraints solver
360        solver.SetConstraints(constraints)
361    from numpy import seterr
362    settings = seterr(all='ignore')
363    solver.Solve(cost, termination)
364    seterr(**settings)
365    soln = solver.bestSolution
366
367    from numpy import ndarray, array
368    if isinstance(soln, ndarray) and not isinstance(guess, ndarray):
369        soln = soln.tolist()
370    elif isinstance(guess, ndarray) and not isinstance(soln, ndarray):
371        soln = array(soln)  #XXX: or always return a list ?
372
373    return soln #XXX: check with 'issolution' ?

626def unique(seq, full=None):
627    """replace the duplicate values with unique values in 'full'
628
629    If full is a type (int or float), then unique values of the given type
630    are selected from range(min(seq),max(seq)). If full is a dict of
631    {'min':min, 'max':max}, then unique floats are selected from
632    range(min(seq),max(seq)). If full is a sequence (list or set), then
633    unique values are selected from the given sequence.
634
635    For example:
636    >>> unique([1,2,3,1,2,4], range(11))
637    [1, 2, 3, 9, 8, 4]
638    >>> unique([1,2,3,1,2,9], range(11))
639    [1, 2, 3, 8, 5, 9]
640    >>> try:
641    ...     unique([1,2,3,1,2,13], range(11))
642    ... except ValueError:
643    ...     pass
644    ...
645    >>>
646    >>> unique([1,2,3,1,2,4], {'min':0, 'max':11})
647    [1, 2, 3, 4.175187820357143, 2.5407265707465716, 4]
648    >>> unique([1,2,3,1,2,4], {'min':0, 'max':11, 'type':int})
649    [1, 2, 3, 6, 8, 4]
650    >>> unique([1,2,3,1,2,4], float)
651    [1, 2, 3, 1.012375036824941, 3.9821250727509905, 4]
652    >>> unique([1,2,3,1,2,10], int)
653    [1, 2, 3, 9, 6, 10]
654    >>> try:
655    ...     unique([1,2,3,1,2,4], int)
656    ... except ValueError:
657    ...     pass
658    ...
659    """
660    unique = set()
661    # replace all duplicates with 'None'
662    seq = [x if x not in unique and not unique.add(x) else None for x in seq]
663    lseq = len(seq)
664    # check type if full not specified
665    if full is None:
666        if all([isinstance(x, int) for x in unique]): full = int
667        else: full = float
668        ok = True
669    else: ok = False
670    # check all unique show up in 'full'
671    if full in (int,float): # specified type, not range
672        ok = ok or full==float or all([isinstance(x, int) for x in unique])
673        msg = "not all items are of type='%s'" % full.__name__
674        _min = min(unique)
675        _max = max(unique)
676    elif isinstance(full, dict): # specified min/max for floats
677        _type = full.pop('type', float) #NOTE: undocumented keys: min,max,type
678        minu = min(unique)
679        maxu = max(unique)
680        _min = full.get('min', minu)
681        _max = full.get('max', maxu)
682        ok = minu >= _min and maxu < _max
683        if not ok:
684            oops = list(unique - set(range(_min,_max)))
685            msg = "x=%s not in %s <= x < %s" % (oops[-1],_min,_max)
686        full = _type
687    else: # full is a list of all possible values
688        ok = unique.issubset(full)
689        if not ok:
690            oops = list(unique - set(full))
691            msg = "%s not in given set" % oops[-1]
692        _min = _max = None
693    if not ok: raise ValueError(msg)
694    # check if a unique sequence is possible to build
695    if full is float:
696        if _min == _max and lseq > 1:
697            msg = "no unique len=%s sequence with %s <= x <= %s" % (lseq,_min,_max)
698            raise ValueError(msg)
699        # replace the 'None' values in seq with 'new' values
700        #XXX: HIGHLY UNLIKELY two numbers will be the same, but possible
701        return [randrange(_min,_max,_int=float) if x is None else x for x in seq]
702    # generate all possible values not found in 'unique'
703    if full is int:
704        if max(lseq - (_max+1 - _min), 0):
705            msg = "no unique len=%s sequence with %s <= x <= %s" % (lseq,_min,_max)
706            raise ValueError(msg)
707        new = list(set(range(_min,_max+1)) - unique)
708    else:
709        if lseq > len(full):
710            msg = "no unique len=%s sequence in given set" % lseq
711            raise ValueError(msg)
712        new = list(set(full) - unique)
713    # ensure randomly ordered
714    shuffle(new)
715    # replace the 'None' values in seq with 'new' values
716    return [new.pop() if x is None else x for x in seq]
