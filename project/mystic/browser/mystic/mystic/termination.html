23class When(tuple):
24  """provide a termination condition with more reporting options.
25
26Terminates when the given condition is satisfied."""
27  def __new__(self, arg):
28    """
29Takes a termination condition:
30    arg    -- termination condition
31
32Usage:
33    >>> from mystic.termination import When, VTR
34    >>> term = When( VTR() )
35    >>> term(solver)  # where solver is a mystic.solver instance"""
36    if isinstance(arg, tuple) and len(arg) == 1: arg = arg[0] # for pickling
37    #XXX: need better filter on inputs
38    if getattr(arg, '__module__', None) != self.__module__:
39      raise TypeError, "'%s' object is not a condition" % arg.__class__.__name__
40    if not getattr(arg, '__len__', None): arg = [arg]
41    return tuple.__new__(self, arg)
42
43  def __call__(self, solver, info=False):
44    """check if the termination conditions are satisfied.
45
46Inputs:
47    solver -- the solver instance
48
49Additional Inputs:
50    info   -- if True, return information about the satisfied conditions"""
51    # return the unsatisfied conditions
52    if info == 'not':
53      return tuple(set([f for f in self if f not in self(solver, 'self')]))
54    # do some filtering...
55    stop = {}
56    [stop.update({f : f(solver, info)}) for f in self]
57    _all = all(stop.values())
58    # return T/F if the conditions are met
59    if not info: return _all
60    # return the satisfied conditions
61    if info == 'self': return tuple(set(stop.keys())) if _all else ()
62    # return info about the satisfied conditions
63    return "; ".join(set("; ".join(stop.values()).split("; "))) if _all else ""
64
65  def __repr__(self):
66    return "When(%s)" % str(self[0])

138def VTR(tolerance=0.005, target=0.0):
139    """cost of last iteration is < tolerance from target:
140
141abs(cost[-1] - target) <= tolerance"""
142    doc = "VTR with %s" % {'tolerance':tolerance, 'target':target}
143    def _VTR(inst, info=False):
144        if info: info = lambda x:x
145        else: info = bool
146        hist = inst.energy_history
147        if not len(hist): return info(null)
148        if abs(hist[-1] - target) <= tolerance: return info(doc)
149        return info(null)
150   #_VTR.__doc__ = "%s(**%s)" % tuple(doc.split(" with "))
151    _VTR.__doc__ = doc
152    return _VTR

190def CandidateRelativeTolerance(xtol=1e-4, ftol=1e-4):
191    """absolute difference in candidates is < tolerance:
192
193abs(xi-x0) <= xtol & abs(fi-f0) <= ftol, where x=params & f=cost"""
194    #NOTE: this termination expects nPop > 1
195    doc = "CandidateRelativeTolerance with %s" % {'xtol':xtol, 'ftol':ftol}
196    def _CandidateRelativeTolerance(inst, info=False):
197        sim = numpy.array(inst.population)
198        fsim = numpy.array(inst.popEnergy)
199        if not len(fsim[1:]):
200            warn = "Warning: Invalid termination condition (nPop < 2)"
201            print warn
202            return warn
203        #   raise ValueError, "Invalid termination condition (nPop < 2)"
204        if info: info = lambda x:x
205        else: info = bool
206        #FIXME: abs(inf - inf) will raise a warning...
207        errdict = numpy.seterr(invalid='ignore') #FIXME: turn off warning
208        answer = max(numpy.ravel(abs(sim[1:]-sim[0]))) <= xtol
209        answer = answer and max(abs(fsim[0]-fsim[1:])) <= ftol
210        numpy.seterr(invalid=errdict['invalid']) #FIXME: turn on warnings
211        if answer: return info(doc)
212        return info(null)
213    _CandidateRelativeTolerance.__doc__ = doc
214    return _CandidateRelativeTolerance

240abs(cost[-1] - fval)/fval <= tolerance *or* (cost[-1] - cost[-g]) = 0 """
241    #NOTE: modified from original behavior

300def GradientNormTolerance(tolerance=1e-5, norm=Inf):
