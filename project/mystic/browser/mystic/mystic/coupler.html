7"""
8Function Couplers
9
10These methods can be used to couple two functions together,
11and represent some common patterns found in applying constraints
12and penalty methods.
13
14For example, the "outer" method called on y = f(x), with outer=c(x),
15will convert y = f(x) to y' = c(f(x)).  Similarly, the "inner" method
16called on y = f(x), with inner=c(x), will convert y = f(x) to  y' = f(c(x)).
17"""
18
19def outer(outer=lambda x:x, args=None, kwds=None):
20    """wrap a function around another function: convert y = f(x) to y' = c(f(x))
21
22This is a useful function for nesting one constraint in another constraint.
23A constraints function takes an iterable x as input, returning a modified x.
24
25    For example:
26    >>> def squared(x):
27    ...   return x**2             
28    ...
29    >>> # equivalent to: ((x+1)**2)
30    >>> @outer(squared)
31    ... def constrain(x):
32    ...   return x+1
33    ...
34    >>> from numpy import array
35    >>> x = array([1,2,3,4,5])
36    >>> constrain(x)
37    array([ 4,  9, 16, 25, 36])
38    """
39    if args is None: args=()
40    if kwds is None: kwds={}
41    def dec(f):
42        def func(x, *argz, **kwdz):
43            return outer(f(x, *argz, **kwdz), *args, **kwds)
44        return func
45    return dec

48def inner(inner=lambda x:x, args=None, kwds=None):
49    """nest a function within another function: convert y = f(x) to y' = f(c(x))
50
51This is a useful function for nesting one constraint in another constraint.
52A constraints function takes an iterable x as input, returning a modified x.
53The "inner" coupler is utilized by mystic.solvers to bind constraints to a cost
54function; thus the constraints are imposed every cost function evaluation.
55
56    For example:
57    >>> def squared(x):
58    ...   return x**2             
59    ...
60    >>> # equivalent to: ((x**2)+1)
61    >>> @inner(squared)
62    ... def constrain(x):
63    ...   return x+1
64    ...
65    >>> from numpy import array
66    >>> x = array([1,2,3,4,5])
67    >>> constrain(x)
68    array([ 2,  5, 10, 17, 26])
69    """
70    if args is None: args=()
71    if kwds is None: kwds={}
72    def dec(f):
73        def func(x, *argz, **kwdz):
74            return f(inner(x, *args, **kwds), *argz, **kwdz)
75        return func
76    return dec

118def additive_proxy(penalty=lambda x:0.0, args=None, kwds=None):
119    """penalize a function with another function: y = f(x) to y' = f(x) + p(x)
120   
121This is useful, for example, in penalizing a cost function where the constraints
122are violated; thus, the satisfying the constraints will be preferred at every
123cost function evaluation.
124
125This function does not preserve decorated function signature, but passes args
126and kwds to the penalty function.
127    """
128    if args is None: args=()
129    if kwds is None: kwds={}
130    def dec(f):
131        def func(x, *argz, **kwdz):
132            return f(x, *args, **kwds) + penalty(x, *argz, **kwdz)
133        return func
134    return dec

136def additive(penalty=lambda x:0.0, args=None, kwds=None):
137    """penalize a function with another function: y = f(x) to y' = f(x) + p(x)
138   
139This is useful, for example, in penalizing a cost function where the constraints
140are violated; thus, the satisfying the constraints will be preferred at every
141cost function evaluation.
142
143    For example:
144    >>> def squared(x):
145    ...   return x**2             
146    ...
147    >>> # equivalent to: (x+1) + (x**2)
148    >>> @additive(squared)
149    ... def constrain(x):
150    ...   return x+1
151    ...
152    >>> from numpy import array
153    >>> x = array([1,2,3,4,5])
154    >>> constrain(x)
155    array([ 3,  7, 13, 21, 31])
156    """
157    if args is None: args=()
158    if kwds is None: kwds={}
159    def dec(f):
160        def func(x, *argz, **kwdz):
161            return f(x, *argz, **kwdz) + penalty(x, *args, **kwds)
162        return func
163    return dec
