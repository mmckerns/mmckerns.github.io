172def line_search_BFGS(f, xk, pk, gfk, old_fval, args=(), c1=1e-4, alpha0=1,\
173                     self=None):
174    """Minimize over alpha, the function ``f(xk+alpha pk)``.
175
176    Uses the interpolation algorithm (Armiijo backtracking) as suggested by
177    Wright and Nocedal in 'Numerical Optimization', 1999, pg. 56-57
178
179    :Returns: (alpha, fc, gc)
180
181    """
182
183    xk = atleast_1d(xk)
184    fc = 0
185    phi0 = old_fval # compute f(xk) -- done in past loop
186    phi_a0 = f(*((xk+alpha0*pk,)+args))
187    fc = fc + 1
188    derphi0 = numpy.dot(gfk,pk)
189
190    if (phi_a0 <= phi0 + c1*alpha0*derphi0):
191        return alpha0, fc, 0, phi_a0
192
193    # Otherwise compute the minimizer of a quadratic interpolant:
194
195    alpha1 = -(derphi0) * alpha0**2 / 2.0 / (phi_a0 - phi0 - derphi0 * alpha0)
196    phi_a1 = f(*((xk+alpha1*pk,)+args))
197    fc = fc + 1
198
199    if (phi_a1 <= phi0 + c1*alpha1*derphi0):
200        return alpha1, fc, 0, phi_a1
201
202    # Otherwise loop with cubic interpolation until we find an alpha which
203    # satifies the first Wolfe condition (since we are backtracking, we will
204    # assume that the value of alpha is not too small and satisfies the second
205    # condition.
206
207    while 1:       # we are assuming pk is a descent direction
208        factor = alpha0**2 * alpha1**2 * (alpha1-alpha0)
209        a = alpha0**2 * (phi_a1 - phi0 - derphi0*alpha1) - \
210            alpha1**2 * (phi_a0 - phi0 - derphi0*alpha0)
211        a = a / factor
212        b = -alpha0**3 * (phi_a1 - phi0 - derphi0*alpha1) + \
213            alpha1**3 * (phi_a0 - phi0 - derphi0*alpha0)
214        b = b / factor
215
216        alpha2 = (-b + numpy.sqrt(abs(b**2 - 3 * a * derphi0))) / (3.0*a)
217        phi_a2 = f(*((xk+alpha2*pk,)+args))
218        fc = fc + 1
219
220        if (phi_a2 <= phi0 + c1*alpha2*derphi0):
221            return alpha2, fc, 0, phi_a2
222
223        if (alpha1 - alpha2) > alpha1 / 2.0 or (1 - alpha2/alpha1) < 0.96:
224            alpha2 = alpha1 / 2.0
225
226        alpha0 = alpha1
227        alpha1 = alpha2
228        phi_a0 = phi_a1
229        phi_a1 = phi_a2
230
231        # In case it gets stuck in this loop, to make 'exit' work.
232        if self._EARLYEXIT:
233            return alpha2, fc, 0, phi_a2

235def approx_fprime(xk,f,epsilon,*args):
236    xk = xk.copy()  #XXX: this line has been added for scipy_ncg
237    f0 = f(*((xk,)+args))
238    grad = numpy.zeros((len(xk),), float)
239    ei = numpy.zeros((len(xk),), float)
240    for k in range(len(xk)):
241        ei[k] = epsilon
242        grad[k] = (f(*((xk+ei,)+args)) - f0)/epsilon
243        ei[k] = 0.0
244    return grad
