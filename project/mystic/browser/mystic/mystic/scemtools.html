37def multinormal_pdf(mean, var):
38    """var must be symmetric positive definite """
39    import numpy
40    vinv = numpy.linalg.inv(var)
41    mu = numpy.array(mean).flatten()
42    n = len(mu)
43
44    # check that var is properly sized
45    dum = numpy.dot(mu,var) - numpy.dot(var,mu)
46
47    prefactor = 1./numpy.sqrt((2 * numpy.pi)**n  * numpy.linalg.det(var))
48    def _(x):
49        xm = numpy.array(x) - mu
50        return prefactor *  numpy.exp(-0.5 * numpy.dot(xm,numpy.dot(vinv,xm)) )
51    return _

53def sequential_deal(inarray, n):
54    """
55- inarray: should be a set of N objects (the objects can be vectors themselves,
56  but inarray should be index-able like a list. It is coerced into a numpy array
57  because the last operations requires that it is also indexable by a 'list.'
58   
59- it should have a length divisble by n, otherwise the reshape will fail
60  (this is a feature !)
61
62- sequential_deal(range(20), 5) wil return a 5 element list, each element being
63  a 4-list of index. (see below)
64
65>>> for l in sequential_deal(range(20),5):
66...    print l
67...
68[ 0  5 10 15]
69[ 1  6 11 16]
70[ 2  7 12 17]
71[ 3  8 13 18]
72[ 4  9 14 19]
73
74    """
75    import numpy
76    cards = numpy.array(inarray)
77    N = len(cards)
78    # this bit of numpy will give, for N=20, n = 5
79    # ord = [ [0,5,10,15], [1,6,11,16], [2,7,12,17], [3,8,13,18], [4,9,14,19] ]
80    ord = numpy.transpose(numpy.array(range(N)).reshape(N/n, n))
81    return [cards[x] for x in ord]

110def sort_complex(c, a):
111    # this is dumb, because c (i.e., a, are almost sorted)
112    # should use the one below instead.
113    import numpy
114    D = zip(a,c)
115    def mycmp(x,y):
116        if x[0] < y[0]:
117            return 1
118        elif x[0] > y[0]:
119            return -1
120        else:
121            return 0
122    D.sort(cmp = mycmp)
123    return numpy.array([x[1] for x in D]), [x[0] for x in D]

125def myinsert(a, x):
126    # a is in descending order
127    from bisect import bisect
128    return bisect([-i for i in a],-x)

130def sort_complex2(c, a):
131    """
132- c and a are partially sorted (either the last one is bad, or the first one)
133- pos : 0 (first one out of order)
134       -1 (last one out of order)
135    """
136    # find where new key is relative to the rest
137    if a[0] < a[1]:
138        ax, cx = a[0], c[0]
139        ind = myinsert(a[1:],ax)
140        a[0:ind], a[ind] = a[1:1+ind], ax
141        c[0:ind], c[ind] = c[1:1+ind], cx
142    elif a[-1] > a[-2]:
143        ax,cx = a[-1], c[-1]
144        ind = myinsert(a[0:-1],ax)
145        a[ind+1:], a[ind] = a[ind:-1], ax
146        c[ind+1:], c[ind] = c[ind:-1], cx
147    return

150    """
151- ak is sorted (descending)
152- Ck[pos] and ak[pos] will be removed, and then
153  c and a spliced in at the proper place
154- pos is 0, or -1
155    """
156    if pos == 0:
157        at = ak[1:]
158    else:
159        at = ak[:-1]
160
161    p = myinsert(at, a)

163    # not done yet.
164    if pos == 0:
165        Ck[0:p], ak[p] = Ck[1:p+1], ak[1:p+1]

167    else:
168        Ck[0:p], ak[p] = Ck[1:p+1], ak[1:p+1]
169        Ck[p], ak[p] = c, a

173def scem(Ck, ak, Sk, Sak, target, cn):
174    """
175This is the SCEM algorithm starting from line [35] of the reference [1].
176
177- [inout] Ck is the kth 'complex' with m points. This should be an m by n array
178  n being the dimensionality of the density function. i.e., the data
179  are arranged in rows.
180
181  Ck is assumed to be sorted according to the target density.
182
183- [inout] ak, the density of the points of Ck.
184
185- [inout] Sk, the entire chain. (should be a list)
186
187- [inout] Sak, the cost of the entire chain (should be a list)
188
189  Sak would be more convenient to use if it is a numpy array, but we need
190  to append to it frequently.
191
192- [in] target: target density function
193
194- [in] cn: jumprate. (see Paragraph 37 of [1.]
195
196- The invariants: ak is always aligned with Ck, and are the cost of Ck
197- Similarly, Sak is always aligned with Sk in the same way.
198
199- On return... sort order in Ck/ak is destroyed. but see sort_complex2
200
201    """
202    import numpy
203    from mystic.tools import random_state
204    prng = random_state(module=numpy.random)
205
206    # function level constants
207    T = 100000. # predefined likelihood ratio. Paragraph 45 of [1.]
208
209    # Sort Ck according to ak
210    #Ck, ak = sort_complex0(Ck, ak)
211    #print "ak before: ", ak
212    #Ck, ak = sort_complex(Ck, ak)
213    sort_complex2(Ck, ak)
214    #print "ak after: ", ak
215
216    # number of points per complex
217    M = Ck.shape[0]
218
219    mu = numpy.mean(Ck,0) # row mean
220
221    # (numpy.cov takes data in columns)
222    Sigma = numpy.cov(numpy.transpose(Ck))
223
224    # Gamma (line 35 of [1]. Best to Worst)
225    Gamma = ak[0] / (ak[-1]+TINY)
226   
227    if len(Sak) >= M:
228        meansak = numpy.mean(Sak[-M:])
229    else:
230        meansak = numpy.mean(Sak)
231    alpha_k = numpy.mean(ak) / (meansak+TINY)
232 
233    if alpha_k < T:
234         # Paragraph 37 of [1]
235         basept = Sk[-1]
236    else: # numpy.mean(asak) is very close to zero.
237         # Paragraph 38 of [1]
238         basept = mu
239
240    # TODO should take a proposal instead !
241    Yt = prng.multivariate_normal(basept, cn*cn * Sigma)
242    cY = target(Yt)   
243
244    # print "new/orig : ", cY, Sak[-1]
245
246    r = min( cY / (Sak[-1]+TINY), 1)
247
248    if prng.rand() <= r:
249        Sk.append(Yt)
250        Sak.append(cY)
251        # Paragraph 43 of [1] (update the best of Ck)
252        Ck[0], ak[0] = Sk[-1], Sak[-1]
253    else:
254        Sk.append(Sk[-1])
255        Sak.append(Sak[-1])
256        if Gamma > T and Sak[-1] > ak[-1]:
257            # Paragraph 43 of [1] (update the worst of Ck)
258            Ck[-1], ak[-1] = Sk[-1], Sak[-1]
259
260    # mainly because of the logic in Paragraph [43] that updates the complex,
261    # this function is not "functional" and modifies its argument instead.
262    # hence no return value
263    return

265def remix(Cs, As):
266    """
267Mixing and dealing the complexes.
268The types of Cs and As are very important....
269    """
270    from mystic.tools import flatten_array
271    q = len(Cs)
272    c2, a2 =  list(flatten_array(Cs,1)), list(flatten_array(As,1))
273    a,b = sort_ab_with_b(c2,a2)
274    C2 = sequential_deal(a, q)
275    A2 = [xx.tolist() for xx in sequential_deal(b, q)]
276    return C2, A2
