14Monitors
15========
16
17Montors provide the ability to monitor progress as the optimization
18is underway. Monitors also can be used to extract and prepare information
19for mystic's analysis viewers. Each of mystic's monitors are customizable,
20and provide the user with a different type of output. The following
21monitors are available::
22    - Monitor        -- the basic monitor; only writes to internal state
23    - LoggingMonitor -- a logging monitor; also writes to a logfile
24    - VerboseMonitor -- a verbose monitor; also writes to stdout/stderr
25    - VerboseLoggingMonitor -- a verbose logging monitor; best of both worlds
26    - CustomMonitor  -- a customizable 'n-variable' version of Monitor
27    - Null           -- a null object, which reliably does nothing
28
29
30Usage
31=====
32
33Typically monitors are either bound to a model function by a modelFactory,
34or bound to a cost function by a Solver.  The typical usage pattern is::
35
36    >>> # get and configure monitors
37    >>> from mystic.monitors import Monitor, VerboseMonitor
38    >>> evalmon = Monitor()
39    >>> stepmon = VerboseMonitor(5)
40    >>>
41    >>> # instantiate and configure the solver
42    >>> from mystic.solvers import NelderMeadSimplexSolver
43    >>> from mystic.termination import CandidateRelativeTolerance as CRT
44    >>> solver = NelderMeadSimplexSolver(len(x0))
45    >>> solver.SetInitialPoints(x0)
46    >>>
47    >>> # associate the monitor with a solver, then solve
48    >>> solver.SetEvaluationMonitor(evalmon)
49    >>> solver.SetGenerationMonitor(stepmon)
50    >>> solver.Solve(rosen, CRT())
51    >>>
52    >>> # access the 'iteration' history
53    >>> stepmon.x     # parameters after each iteration
54    >>> stepmon.y     # cost after each iteration
55    >>>
56    >>> # access the 'evaluation' history
57    >>> evalmon.x     # parameters after each evaluation
58    >>> evalmon.y     # cost after each evaluation
59
60

195        "avoid double-conversion by combining k's"

211    def _k(self, y, type=list):
212        return multiply(y, self.k, type)

214    def _ik(self, y, k=False, type=list):
215        if k: return y # k's already applied, so don't un-apply it
216        return divide(y, self.k, type)

228class VerboseMonitor(Monitor):
229    """A verbose version of the basic Monitor.
230
231Prints ChiSq every 'interval', and optionally prints
232current parameters every 'xinterval'.
233    """
234    def __init__(self, interval=10, xinterval=numpy.inf, all=True, **kwds):
235        super(VerboseMonitor,self).__init__(**kwds)
236        if not interval or interval is numpy.nan: interval = numpy.inf
237        if not xinterval or xinterval is numpy.nan: xinterval = numpy.inf
238        self._yinterval = interval
239        self._xinterval = xinterval
240        self._all = all
241        return
242    def info(self, message):
243        super(VerboseMonitor,self).info(message)
244        print "%s" % "".join(["",str(message)])
245        return
246    def __call__(self, x, y, id=None, best=0, k=False):
247        super(VerboseMonitor,self).__call__(x, y, id, k=k)
248        if self._yinterval is not numpy.inf and \
249           int((self._step-1) % self._yinterval) == 0:
250            if not list_or_tuple_or_ndarray(y):
251                who = ''
252                y = " %f" % self._ik(self._y[-1], k)
253            elif self._all:
254                who = ''
255                y = " %s" % self._ik(self._y[-1], k)
256            else:
257                who = ' best'
258                y = " %f" % self._ik(self._y[-1][best], k)
259            msg = "Generation %d has%s Chi-Squared:%s" % (self._step-1,who,y)
260            if id is not None: msg = "[id: %d] " % (id) + msg
261            print msg
262        if self._xinterval is not numpy.inf and \
263           int((self._step-1) % self._xinterval) == 0:
264            if not list_or_tuple_or_ndarray(x):
265                who = ''
266                x = " %f" % self._x[-1]
267            elif self._all:
268                who = ''
269                x = "\n %s" % self._x[-1]
270            else:
271                who = ' best'
272                x = "\n %s" % self._x[-1][best]
273            msg = "Generation %d has%s fit parameters:%s" % (self._step-1,who,x)
274            if id is not None: msg = "[id: %d] " % (id) + msg
275            print msg
276        return

279class LoggingMonitor(Monitor):
280    """A basic Monitor that writes to a file at specified intervals.
281
282Logs ChiSq and parameters to a file every 'interval'
283    """
284    def __init__(self, interval=1, filename='log.txt', new=False, all=True, info=None, **kwds):
285        import datetime
286        super(LoggingMonitor,self).__init__(**kwds)
287        self._filename = filename
288        if not interval or interval is numpy.nan: interval = numpy.inf
289        self._yinterval = interval
290        self._xinterval = interval
291        if new: ind = 'w'
292        else: ind = 'a'
293        self._file = open(self._filename,ind)
294        self._file.write("# %s\n" % datetime.datetime.now().ctime() )
295        if info: self._file.write("# %s\n" % str(info))
296        self._file.write("# ___#___  __ChiSq__  __params__\n")
297        self._file.close()
298        self._all = all
299        return
300    def info(self, message):
301        super(LoggingMonitor,self).info(message)
302        self._file = open(self._filename,'a')
303        self._file.write("# %s\n" % str(message))
304        self._file.close()
305        return
306    def __call__(self, x, y, id=None, best=0, k=False):
307        self._file = open(self._filename,'a')
308        super(LoggingMonitor,self).__call__(x, y, id, k=k)
309        if self._yinterval is not numpy.inf and \
310           int((self._step-1) % self._yinterval) == 0:
311            if not list_or_tuple_or_ndarray(y):
312                y = "%f" % self._ik(self._y[-1], k)
313            elif self._all:
314                y = "%s" % self._ik(self._y[-1], k)
315            else:
316                y = "%f" % self._ik(self._y[-1][best], k)
317            if not list_or_tuple_or_ndarray(x):
318                x = "[%f]" % self._x[-1]
319            elif self._all:
320                xa = self._x[-1]
321                if not list_or_tuple_or_ndarray(xa):
322                  x = "[%f]" % xa
323                else:
324                  x = "%s" % xa
325            else:
326                xb = self._x[-1][best]
327                if not list_or_tuple_or_ndarray(xb):
328                  x = "[%f]" % xb
329                else:
330                  x = "%s" % xb
331            step = [self._step-1]
332            if id is not None: step.append(id)
333            self._file.write("  %s     %s   %s\n" % (tuple(step), y, x))
334        self._file.close()
335        return
336    def __reduce__(self):
337        interval = self._yinterval       
338        filename = self._filename
339        new = False
340        all = self._all
341        info = None
342        args = (interval, filename, new, all, info)
343        k = self.k
344        state = dict(_x=self._x,_y=self._y,_id=self._id,_info=self._info,k=k)
345        return (self.__class__, args, state)
346    def __setstate__(self, state):
347        self.__dict__.update(state)
348        return
349    pass
