23   #    seeker - an optimiztion algorithm
24   #    map - an enselble map
25   #    traj - trajectory configuration (bool)
26   #    disp - verbosity configuration (bool)

30   #    _memoize - apply caching archive to sprayer

33   #    UseTrajectories - save all sprayers, thus save all trajectories
34   #    Reset - clear caching archive and saved sprayers (thus trajectories)
35   #    Trajectories - fetch (step, param, cost) for all solver trajectories
36   #    Samples - all sampled points as a n-D array (output is array[-1])
37   #    Values - fetch values (model(x)) for sampled points on surface (model)
38   #    Coordinates - fetch inputs (x) for sampled points on surface (model)
39   #    Minima - fetch {x: model(x)} for all x that minimize model
40   #    _summarize - print diagnostics on the value/quantity of solved values
41   #
42   # searcher (or sprayer) can:
43   #    _print - print bestSolution and bestEnergy for each sprayer
44   #    _configure (model, bounds, stop, monitor) - configure sprayer
45   #    _solve - spray multiple seekers

148    def Search(self, model, bounds, stop=None, monitor=None, traj=None, disp=None):
149        self.traj = self.traj if traj is None else traj
150        self.disp = self.disp if disp is None else disp
151        self._configure(model, bounds, stop, monitor)
152        count = 0 if self.retry else -1 #XXX: 'rerun' much shorter... unless clear
153        sid = 0  # keep track of which solver is which across multiple runs
154        while self.retry > count: # stop after retry consecutive no new results
155            _size = -1
156            size = osize = len(self.archive) #XXX: compare 'size' or 'len(vals)'?
157            while size > _size: # stop if no new results
158                _size = size
159                sid, size = self._search(sid) # uses self.traj and self.disp
160            if size == osize: count = count + 1
161            else: count = 0

163        #NOTE: traj & disp are sticky
