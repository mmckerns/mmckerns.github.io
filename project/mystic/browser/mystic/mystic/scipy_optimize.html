23
24This module contains a collection of optimization routines adapted

29Minimal function interface to optimization routines::
30   fmin        -- Nelder-Mead Simplex algorithm
31                    (uses only function calls)
32   fmin_powell -- Powell's (modified) level set method (uses only
33                    function calls)

36   NelderMeadSimplexSolver -- Nelder-Mead Simplex algorithm

40   - EvaluationMonitor = Monitor()
41   - StepMonitor = Monitor()
42   - termination = CandidateRelativeTolerance(xtol,ftol)

45   - EvaluationMonitor = Monitor()
46   - StepMonitor = Monitor()
47   - termination = NormalizedChangeOverGeneration(ftol)

51=====
52

100    def _setSimplexWithinRangeBoundary(self, radius=None):
101        """ensure that initial simplex is set within bounds
102    - radius: size of the initial simplex [default=0.05]"""
103        x0 = self.population[0]
104        #code modified from park-1.2/park/simplex.py (version 1257)
105        if self._useStrictRange:
106            x0 = self._clipGuessWithinRangeBoundary(x0)
107
108        if radius is None: radius = 0.05 # nonzdelt=0.05 from scipy-0.9
109        val = x0*(1+radius)
110        val[val==0] = (radius**2) * 0.1 # zdelt=0.00025 update from scipy-0.9
111        if not self._useStrictRange:
112            self.population[0] = x0
113            return val
114
115        lo = self._strictMin
116        hi = self._strictMax
117        radius = clip(radius,0,0.5)
118        # rescale val by bounded range...
119        # (increases fit for tight bounds; makes worse[?] for large bounds)
120        bounded = ~numpy.isinf(lo) & ~numpy.isinf(hi)
121        val[bounded] = x0[bounded] + (hi[bounded]-lo[bounded])*radius
122        # crop val at bounds
123        settings = numpy.seterr(all='ignore')
124        val[val<lo] = lo[val<lo]
125        val[val>hi] = hi[val>hi]
126        numpy.seterr(**settings)
127        # handle collisions (when val[i] == x0[i])
128        collision = val==x0
129        if numpy.any(collision):
130            rval = x0*(1-radius)
131            rval[rval==0] = -radius
132            rval[bounded] = x0[bounded] - (hi[bounded]-lo[bounded])*radius
133            val[collision] = rval[collision]
134        # make tolerance relative for bounded parameters
135     #  tol = numpy.ones(x0.shape)*xtol
136     #  tol[bounded] = (hi[bounded]-lo[bounded])*xtol
137     #  xtol = tol
138        self.population[0] = x0
139        return val

145    def _decorate_objective(self, cost, ExtraArgs=None):

147        #print ("@", cost, ExtraArgs, max)
148        raw = cost

150        self._fcalls, cost = wrap_function(cost, ExtraArgs, self._evalmon)
151        if self._useStrictRange:
152            if self.generations:
153                #NOTE: pop[0] was best, may not be after resetting simplex
154                for i,j in enumerate(self._setSimplexWithinRangeBoundary()):

162        if self._reducer:

164            cost = reduced(self._reducer, arraylike=True)(cost)
165        # hold on to the 'wrapped' and 'raw' cost function
166        self._cost = (cost, raw, ExtraArgs)

170    def _Step(self, cost=None, ExtraArgs=None, **kwds):
171        """perform a single optimization iteration
172        Note that ExtraArgs should be a *tuple* of extra arguments"""
173        # process and activate input settings
174        settings = self._process_inputs(kwds)
175        for key in settings:
176            exec "%s = settings['%s']" % (key,key)
177
178        # HACK to enable not explicitly calling _decorate_objective
179        cost = self._bootstrap_objective(cost, ExtraArgs)
180
181        rho = 1; chi = 2; psi = 0.5; sigma = 0.5;
182        init = False  # flag to do 0th iteration 'post-initialization'
183
184        if not len(self._stepmon): # do generation = 0
185            init = True
186            x0 = self.population[0]
187            x0 = asfarray(x0).flatten()
188            x0 = asfarray(self._constraints(x0))
189            #####XXX: this blows away __init__, so replace __init__ with this?
190            N = len(x0)
191            rank = len(x0.shape)
192            if not -1 < rank < 2:
193                raise ValueError, "Initial guess must be a scalar or rank-1 sequence."
194            if rank == 0:
195                sim = numpy.zeros((N+1,), dtype=x0.dtype)
196            else:
197                sim = numpy.zeros((N+1,N), dtype=x0.dtype)
198            fsim = numpy.ones((N+1,), float) * self._init_popEnergy
199            ####################################################
200            sim[0] = x0
201            fsim[0] = cost(x0)
202
203        elif not self.generations: # do generations = 1
204            #--- ensure initial simplex is within bounds ---
205            val = self._setSimplexWithinRangeBoundary(radius)
206            #--- end bounds code ---
207            sim = self.population
208            fsim = self.popEnergy
209            x0 = sim[0]
210            N = len(x0)
211            # populate the simplex
212            for k in range(0,N):
213                y = numpy.array(x0,copy=True)
214                y[k] = val[k]
215                sim[k+1] = y
216                f = cost(y) #XXX: use self._map?
217                fsim[k+1] = f
218
219        else: # do generations > 1
220            sim = self.population
221            fsim = self.popEnergy
222            N = len(sim[0])
223            one2np1 = range(1,N+1)
224
225            # apply constraints  #XXX: is this the only appropriate place???
226            sim[0] = asfarray(self._constraints(sim[0]))
227
228            xbar = numpy.add.reduce(sim[:-1],0) / N
229            xr = (1+rho)*xbar - rho*sim[-1]
230            fxr = cost(xr)
231            doshrink = 0
232
233            if fxr < fsim[0]:
234                xe = (1+rho*chi)*xbar - rho*chi*sim[-1]
235                fxe = cost(xe)
236
237                if fxe < fxr:
238                    sim[-1] = xe
239                    fsim[-1] = fxe
240                else:
241                    sim[-1] = xr
242                    fsim[-1] = fxr
243            else: # fsim[0] <= fxr
244                if fxr < fsim[-2]:
245                    sim[-1] = xr
246                    fsim[-1] = fxr
247                else: # fxr >= fsim[-2]
248                    # Perform contraction
249                    if fxr < fsim[-1]:
250                        xc = (1+psi*rho)*xbar - psi*rho*sim[-1]
251                        fxc = cost(xc)
252   
253                        if fxc <= fxr:
254                            sim[-1] = xc
255                            fsim[-1] = fxc
256                        else:
257                            doshrink=1
258                    else:
259                        # Perform an inside contraction
260                        xcc = (1-psi)*xbar + psi*sim[-1]
261                        fxcc = cost(xcc)
262
263                        if fxcc < fsim[-1]:
264                            sim[-1] = xcc
265                            fsim[-1] = fxcc
266                        else:
267                            doshrink = 1
268
269                    if doshrink:
270                        for j in one2np1:
271                            sim[j] = sim[0] + sigma*(sim[j] - sim[0])
272                            fsim[j] = cost(sim[j]) #XXX: use self._map?
273
274        if len(self._stepmon):
275            # sort so sim[0,:] has the lowest function value
276            ind = numpy.argsort(fsim)
277            sim = numpy.take(sim,ind,0)
278            fsim = numpy.take(fsim,ind,0)
279        self.population = sim # bestSolution = sim[0]
280        self.popEnergy = fsim # bestEnergy = fsim[0]
281        self._stepmon(sim[0], fsim[0], self.id) # sim = all; "best" is sim[0]
282        # if savefrequency matches, then save state
283        self._AbstractSolver__save_state()
284
285        # do callback
286        if callback is not None: callback(self.bestSolution)
287        # initialize termination conditions, if needed
288        if init: self._termination(self) #XXX: at generation 0 or always?
289        return #XXX: call Terminated ?

335def fmin(cost, x0, args=(), bounds=None, xtol=1e-4, ftol=1e-4,
336         maxiter=None, maxfun=None, full_output=0, disp=1, retall=0,
337         callback=None, **kwds):
338    """Minimize a function using the downhill simplex algorithm.
339   
340Description:
341
342    Uses a Nelder-Mead simplex algorithm to find the minimum of
343    a function of one or more variables. Mimics the scipy.optimize.fmin
344    interface.
345
346Inputs:
347
348    cost -- the Python function or method to be minimized.
349    x0 -- ndarray - the initial guess.
350
351Additional Inputs:
352
353    args -- extra arguments for cost.
354    bounds -- list - n pairs of bounds (min,max), one pair for each parameter.
355    xtol -- number - acceptable relative error in xopt for convergence.
356    ftol -- number - acceptable relative error in cost(xopt) for
357        convergence.
358    maxiter -- number - the maximum number of iterations to perform.
359    maxfun -- number - the maximum number of function evaluations.
360    full_output -- number - non-zero if fval and warnflag outputs are
361        desired.
362    disp -- number - non-zero to print convergence messages.
363    retall -- number - non-zero to return list of solutions at each
364        iteration.
365    callback -- an optional user-supplied function to call after each
366        iteration.  It is called as callback(xk), where xk is the
367        current parameter vector.
368    handler -- boolean - enable/disable handling of interrupt signal.
369    itermon -- monitor - override the default GenerationMonitor.
370    evalmon -- monitor - override the default EvaluationMonitor.
371    constraints -- an optional user-supplied function.  It is called as
372        constraints(xk), where xk is the current parameter vector.
373        This function must return xk', a parameter vector that satisfies
374        the encoded constraints.
375    penalty -- an optional user-supplied function.  It is called as
376        penalty(xk), where xk is the current parameter vector.
377        This function should return y', with y' == 0 when the encoded
378        constraints are satisfied, and y' > 0 otherwise.
379
380Returns: (xopt, {fopt, iter, funcalls, warnflag}, {allvecs})
381
382    xopt -- ndarray - minimizer of function
383    fopt -- number - value of function at minimum: fopt = cost(xopt)
384    iter -- number - number of iterations
385    funcalls -- number - number of function calls
386    warnflag -- number - Integer warning flag:
387        1 : 'Maximum number of function evaluations.'
388        2 : 'Maximum number of iterations.'
389    allvecs -- list - a list of solutions at each iteration
390
391    """
392    handler = kwds['handler'] if 'handler' in kwds else False
393
394    from mystic.monitors import Monitor
395    stepmon = kwds['itermon'] if 'itermon' in kwds else Monitor()
396    evalmon = kwds['evalmon'] if 'evalmon' in kwds else Monitor()
397
398    if xtol: #if tolerance in x is provided, use CandidateRelativeTolerance
399        from mystic.termination import CandidateRelativeTolerance as CRT
400        termination = CRT(xtol,ftol)
401    else:
402        from mystic.termination import VTRChangeOverGeneration
403        termination = VTRChangeOverGeneration(ftol)
404    solver = NelderMeadSimplexSolver(len(x0))
405    solver.SetInitialPoints(x0)
406    solver.SetEvaluationLimits(maxiter,maxfun)
407    solver.SetEvaluationMonitor(evalmon)
408    solver.SetGenerationMonitor(stepmon)
409    if 'penalty' in kwds:
410        solver.SetPenalty(kwds['penalty'])
411    if 'constraints' in kwds:
412        solver.SetConstraints(kwds['constraints'])
413    if bounds is not None:
414        minb,maxb = unpair(bounds)
415        solver.SetStrictRanges(minb,maxb)
416
417    if handler: solver.enable_signal_handler()
418    solver.Solve(cost, termination=termination, \
419                 disp=disp, ExtraArgs=args, callback=callback)
420    solution = solver.Solution()
421
422    # code below here pushes output to scipy.optimize.fmin interface
423   #x = list(solver.bestSolution)
424    x = solver.bestSolution
425    fval = solver.bestEnergy
426    warnflag = 0
427    fcalls = solver.evaluations
428    iterations = solver.generations
429    allvecs = stepmon.x
430
431    if fcalls >= solver._maxfun:
432        warnflag = 1
433    elif iterations >= solver._maxiter:
434        warnflag = 2
435
436    if full_output:
437        retlist = x, fval, iterations, fcalls, warnflag
438        if retall:
439            retlist += (allvecs,)
440    else:
441        retlist = x
442        if retall:
443            retlist = (x, allvecs)
444
445    return retlist

449def _linesearch_powell(func, p, xi, tol=1e-3):
450    # line-search algorithm using fminbound
451    #  find the minimium of the function
452    #  func(x0+ alpha*direc)
453    def myfunc(alpha):
454        return func(p + alpha * xi)
455    settings = numpy.seterr(all='ignore')
456    alpha_min, fret, iter, num = brent(myfunc, full_output=1, tol=tol)
457    numpy.seterr(**settings)
458    xi = alpha_min*xi
459    return squeeze(fret), p+xi, xi

509        if not len(self._stepmon): # do generation = 0
510            init = True
511            x = asfarray(x).flatten()
512            x = asfarray(self._constraints(x))
513            N = len(x) #XXX: this should be equal to self.nDim
514            rank = len(x.shape)
515            if not -1 < rank < 2:
516                raise ValueError, "Initial guess must be a scalar or rank-1 sequence."
517
518            if direc is None:
519                direc = eye(N, dtype=float)
520            else:
521                direc = asarray(direc, dtype=float)
522            fval = squeeze(cost(x))
523            if self._maxiter != 0:
524                self._stepmon(x, fval, self.id) # get initial values
525                # if savefrequency matches, then save state
526                self._AbstractSolver__save_state()
527
528        elif not self.generations: # do generations = 1
529            ilist = range(len(x))
530            x1 = x.copy()
531            # do initial "second half" of solver step
532            fx = fval
533            bigind = 0
534            delta = 0.0
535            for i in ilist:
536                direc1 = self._direc[i]
537                fx2 = fval
538                fval, x, direc1 = _linesearch_powell(cost, x, direc1, tol=xtol*100)
539                if (fx2 - fval) > delta:
540                    delta = fx2 - fval
541                    bigind = i
542
543                # apply constraints
544                x = asfarray(self._constraints(x)) #XXX: use self._map?
545            # decouple from 'best' energy
546            self.energy_history = self.energy_history + [fval]
547
548        else: # do generations > 1
549            # Construct the extrapolated point
550            direc1 = x - x1
551            x2 = 2*x - x1
552            x1 = x.copy()
553            fx2 = squeeze(cost(x2))
554
555            if (fx > fx2):
556                t = 2.0*(fx+fx2-2.0*fval)
557                temp = (fx-fval-delta)
558                t *= temp*temp
559                temp = fx-fx2
560                t -= delta*temp*temp
561                if t < 0.0:
562                    fval, x, direc1 = _linesearch_powell(cost, x, direc1, tol=xtol*100)
563                    direc[bigind] = direc[-1]
564                    direc[-1] = direc1
565
566           #        x = asfarray(self._constraints(x))
567
568            self._direc = direc
569            self.population[0] = x   # bestSolution
570            self.popEnergy[0] = fval # bestEnergy
571            self.energy_history = None # resync with 'best' energy
572            self._stepmon(x, fval, self.id) # get ith values
573            # if savefrequency matches, then save state
574            self._AbstractSolver__save_state()
575
576            fx = fval
577            bigind = 0
578            delta = 0.0
579            ilist = range(len(x))
580            for i in ilist:
581                direc1 = direc[i]
582                fx2 = fval
583                fval, x, direc1 = _linesearch_powell(cost, x, direc1, tol=xtol*100)
584                if (fx2 - fval) > delta:
585                    delta = fx2 - fval
586                    bigind = i
587
588                # apply constraints
589                x = asfarray(self._constraints(x)) #XXX: use self._map?
590
591            # decouple from 'best' energy
592            self.energy_history = self.energy_history + [fval]
593
594        self.__internals = [x1, fx, bigind, delta]
595        self._direc = direc
596        self.population[0] = x   # bestSolution
597        self.popEnergy[0] = fval # bestEnergy
598
599        # do callback
600        if callback is not None: callback(self.bestSolution)
601        # initialize termination conditions, if needed
602        if init: self._termination(self) #XXX: at generation 0 or always?
603        return #XXX: call Terminated ?

605    def Finalize(self, **kwds):
606        """cleanup upon exiting the main optimization loop"""
607        if self.energy_history != None and self._live:
608            self.energy_history = None # resync with 'best' energy
609            self._stepmon(self.bestSolution, self.bestEnergy, self.id)
610            # if savefrequency matches, then save state
611            self._AbstractSolver__save_state()
612        self._live = False
613        return

667                maxiter=None, maxfun=None, full_output=0, disp=1, retall=0,
668                callback=None, direc=None, **kwds):
669    """Minimize a function using modified Powell's method.
670   
671Description:
672
673    Uses a modified Powell Directional Search algorithm to find
674    the minimum of function of one or more variables.  Mimics the
675    scipy.optimize.fmin_powell interface.
676
677Inputs:
678
679    cost -- the Python function or method to be minimized.
680    x0 -- ndarray - the initial guess.
681
682Additional Inputs:
683
684    args -- extra arguments for cost.
685    bounds -- list - n pairs of bounds (min,max), one pair for each parameter.
686    xtol -- number - acceptable relative error in xopt for
687        convergence.
688    ftol -- number - acceptable relative error in cost(xopt) for
689        convergence.
690    gtol -- number - maximum number of iterations to run without improvement.
691    maxiter -- number - the maximum number of iterations to perform.
692    maxfun -- number - the maximum number of function evaluations.
693    full_output -- number - non-zero if fval and warnflag outputs
694        are desired.
695    disp -- number - non-zero to print convergence messages.
696    retall -- number - non-zero to return list of solutions at each
697        iteration.
698    callback -- an optional user-supplied function to call after each
699        iteration.  It is called as callback(xk), where xk is the
700        current parameter vector.
701    direc -- initial direction set.
702    handler -- boolean - enable/disable handling of interrupt signal.
703    itermon -- monitor - override the default GenerationMonitor.
704    evalmon -- monitor - override the default EvaluationMonitor.
705    constraints -- an optional user-supplied function.  It is called as
706        constraints(xk), where xk is the current parameter vector.
707        This function must return xk', a parameter vector that satisfies
708        the encoded constraints.
709    penalty -- an optional user-supplied function.  It is called as
710        penalty(xk), where xk is the current parameter vector.
711        This function should return y', with y' == 0 when the encoded
712        constraints are satisfied, and y' > 0 otherwise.

714Returns: (xopt, {fopt, iter, funcalls, warnflag, direc}, {allvecs})

716    xopt -- ndarray - minimizer of function
717    fopt -- number - value of function at minimum: fopt = cost(xopt)
718    iter -- number - number of iterations
719    funcalls -- number - number of function calls
720    warnflag -- number - Integer warning flag:
721        1 : 'Maximum number of function evaluations.'
722        2 : 'Maximum number of iterations.'
723    direc -- current direction set
724    allvecs -- list - a list of solutions at each iteration
725
726    """
727    #FIXME: need to resolve "direc"
728    #        - should just pass 'direc', and then hands-off ?  How return it ?
729
730    handler = kwds['handler'] if 'handler' in kwds else False
731
732    from mystic.monitors import Monitor
733    stepmon = kwds['itermon'] if 'itermon' in kwds else Monitor()
734    evalmon = kwds['evalmon'] if 'evalmon' in kwds else Monitor()
