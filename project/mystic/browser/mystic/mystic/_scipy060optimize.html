306def _cubicmin(a,fa,fpa,b,fb,c,fc):
307    # finds the minimizer for a cubic polynomial that goes through the
308    #  points (a,fa), (b,fb), and (c,fc) with derivative at a of fpa.
309    #
310    # if no minimizer can be found return None
311    #
312    # f(x) = A *(x-a)^3 + B*(x-a)^2 + C*(x-a) + D
313
314    C = fpa
315    D = fa
316    db = b-a
317    dc = c-a
318    if (db == 0) or (dc == 0) or (b==c): return None
319    denom = (db*dc)**2 * (db-dc)
320    d1 = empty((2,2))
321    d1[0,0] = dc**2
322    d1[0,1] = -db**2
323    d1[1,0] = -dc**3
324    d1[1,1] = db**3
325    [A,B] = numpy.dot(d1,asarray([fb-fa-C*db,fc-fa-C*dc]).flatten())
326    A /= denom
327    B /= denom
328    radical = B*B-3*A*C
329    if radical < 0:  return None
330    if (A == 0): return None
331    xmin = a + (-B + sqrt(radical))/(3*A)
332    return xmin

335def _quadmin(a,fa,fpa,b,fb):
336    # finds the minimizer for a quadratic polynomial that goes through
337    #  the points (a,fa), (b,fb) with derivative at a of fpa
338    # f(x) = B*(x-a)^2 + C*(x-a) + D
339    D = fa
340    C = fpa
341    db = b-a*1.0
342    if (db==0): return None
343    B = (fb-D-C*db)/(db*db)
344    if (B <= 0): return None
345    xmin = a  - C / (2.0*B)
346    return xmin

527        i = i + 1
528        alpha0 = alpha1
529        alpha1 = alpha2
530        phi_a0 = phi_a1
531        phi_a1 = phi(alpha1)
532        derphi_a0 = derphi_a1

551def line_search_BFGS(f, xk, pk, gfk, old_fval, args=(), c1=1e-4, alpha0=1):
552    """Minimize over alpha, the function f(xk+alpha pk)
553
554    Uses the interpolation algorithm (Armiijo backtracking) as suggested by
555    Wright and Nocedal in 'Numerical Optimization', 1999, pg. 56-57
556
557    :Returns: (alpha, fc, gc)
558    """
559   
560    xk = atleast_1d(xk)
561    fc = 0
562    phi0 = old_fval                            # compute f(xk) -- done in past loop
563    phi_a0 = f(*((xk+alpha0*pk,)+args))     
564    fc = fc + 1
565    derphi0 = numpy.dot(gfk,pk)
566
567    if (phi_a0 <= phi0 + c1*alpha0*derphi0):
568        return alpha0, fc, 0, phi_a0
569
570    # Otherwise compute the minimizer of a quadratic interpolant:
571
572    alpha1 = -(derphi0) * alpha0**2 / 2.0 / (phi_a0 - phi0 - derphi0 * alpha0)
573    phi_a1 = f(*((xk+alpha1*pk,)+args))
574    fc = fc + 1
575
576    if (phi_a1 <= phi0 + c1*alpha1*derphi0):
577        return alpha1, fc, 0, phi_a1
578
579    # Otherwise loop with cubic interpolation until we find an alpha which
580    # satifies the first Wolfe condition (since we are backtracking, we will
581    # assume that the value of alpha is not too small and satisfies the second
582    # condition.
583
584    while 1:       # we are assuming pk is a descent direction
585        factor = alpha0**2 * alpha1**2 * (alpha1-alpha0)
586        a = alpha0**2 * (phi_a1 - phi0 - derphi0*alpha1) - \
587            alpha1**2 * (phi_a0 - phi0 - derphi0*alpha0)
588        a = a / factor
589        b = -alpha0**3 * (phi_a1 - phi0 - derphi0*alpha1) + \
590            alpha1**3 * (phi_a0 - phi0 - derphi0*alpha0)
591        b = b / factor
592
593        alpha2 = (-b + numpy.sqrt(abs(b**2 - 3 * a * derphi0))) / (3.0*a)
594        phi_a2 = f(*((xk+alpha2*pk,)+args))
595        fc = fc + 1
596
597        if (phi_a2 <= phi0 + c1*alpha2*derphi0):
598            return alpha2, fc, 0, phi_a2
599
600        if (alpha1 - alpha2) > alpha1 / 2.0 or (1 - alpha2/alpha1) < 0.96:
601            alpha2 = alpha1 / 2.0
602
603        alpha0 = alpha1
604        alpha1 = alpha2
605        phi_a0 = phi_a1
606        phi_a1 = phi_a2
