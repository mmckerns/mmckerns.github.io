8"""
9This module contains classes that aid in constructing cost functions.
10Cost function can easily be created by hand; however, mystic also
11provides an automated method that allows the dynamic wrapping of
12forward models into cost function objects.
13
14Usage
15=====
16
17The basic usage pattern for a cost factory is to generate a cost function
18from a set of data points and a corresponding set of evaluation points.
19The cost factory requires a "model factory", which is just a generator
20of model function instances from a list of coefficients. The following example
21uses numpy.poly1d, which provides a factory for generating polynomials. An
22expanded version of the following can be found in `mystic.examples.example12`.
23
24    >>> # get a model factory
25    >>> import numpy as np
26    >>> FunctionFactory = np.poly1d
27    >>>
28    >>> # generate some evaluation points
29    >>> xpts = 0.1 * np.arange(-50.,51.)
30    >>>
31    >>> # we don't have real data, so generate fake data from target and model
32    >>> target = [2.,-5.,3.]
33    >>> ydata = FunctionFactory(target)(xpts)
34    >>> noise = np.random.normal(0,1,size=len(ydata))
35    >>> ydata = ydata + noise
36    >>>
37    >>> # get a cost factory
38    >>> from mystic.forward_model import CostFactory
39    >>> C = CostFactory()
40    >>>
41    >>> # generate a cost function for the model factory
42    >>> metric = lambda x: np.sum(x*x)
43    >>> C.addModel(FunctionFactory, inputs=len(target))
44    >>> cost = C.getCostFunction(evalpts=xpts, observations=ydata,
45    ...                                        sigma=1.0, metric=metric)
46    >>>
47    >>> # pass the cost function to the optimizer
48    >>> initial_guess = [1.,-2.,1.]
49    >>> solution = fmin_powell(cost, initial_guess)
50    >>> print solution
51    [ 2.00495233 -5.0126248   2.72873734]
52
53
54In general, a user will be required to write their own model factory.
55See the examples contained in `mystic.models` for more information.
56
57The CostFactory can be used to couple models together into a single cost
58function. For an example, see `mystic.examples.forward_model`.
59
60"""

141    def getForwardEvaluator(self, evalpts):
142        """
143Get a model factory that allows simultaneous evaluation of all forward models
144for the same set of evaluation points.
145
146Inputs:
147    evalpts -- a list of evaluation points
148
149Example:
150    >>> import numpy as np
151    >>> C = CostFactory()
152    >>> C.addModel(np.poly, inputs=3)
153    >>> F = C.getForwardEvaluator([1,2,3,4,5])
154    >>> F([1,0,0])
155    [array([ 1,  4,  9, 16, 25])]
156    >>> F([0,1,0])
157    [array([1, 2, 3, 4, 5])]
158        """
159        #NOTE: does NOT go through inputChecker
160        def _(params):
161            out = []
162            ind = 0
163            for F, n, ofilt in zip(self._forwardFactories, self._inputs, self._outputFilters):
164                m = F(params[ind:ind+n])
165                out.append(ofilt(m(evalpts)))
166                ind = ind+n
167            return out
168        return _

170    def getVectorCostFunction(self, evalpts, observations):
171        """
172Get a vector cost function that allows simultaneous evaluation of all
173forward models for the same set of evaluation points and observation points.
174
175Inputs:
176    evalpts -- a list of evaluation points
177    observations -- a list of data points
178
179The vector cost metric is hard-wired to be the sum of the difference of
180getForwardEvaluator(evalpts) and the observations.
181
182NOTE: Input parameters do NOT go through filters registered as inputCheckers.
183
184Example:
185    >>> import numpy as np
186    >>> C = CostFactory()
187    >>> C.addModel(np.poly, inputs=3)
188    >>> x = np.array([-2., -1., 0., 1., 2.])
189    >>> y = np.array([-4., -2., 0., 2., 4.])
190    >>> F = C.getVectorCostFunction(x, y)
191    >>> F([1,0,0])
192    0.0
193    >>> F([2,0,0])
194    10.0
195        """
196        def _(params):
197            forward = self.getForwardEvaluator(evalpts)
198            return sum(forward(params) - observations)
199        return _

202        """
203Get a cost function that allows simultaneous evaluation of all forward models
204for the same set of evaluation points and observation points.
205

207    evalpts -- a list of evaluation points

214be passed in.
215

217

224    >>> F = C.getCostFunction(x, y, metric=lambda x: np.sum(x))

226    0.0
227    >>> F([2,0,0])

260    def getCostFunctionSlow(self, evalpts, observations):
261        """
262Get a cost function that allows simultaneous evaluation of all forward models
263for the same set of evaluation points and observation points.
264
265Inputs:
266    evalpts -- a list of evaluation points
267    observations -- a list of data points
268
269The cost metric is hard-wired to be the sum of the real part of |x|^2,
270where x is the VectorCostFunction for a given set of parameters.
271
272NOTE: Input parameters do NOT go through filters registered as inputCheckers.
273
274Example:
275    >>> import numpy as np
276    >>> C = CostFactory()
277    >>> C.addModel(np.poly, inputs=3)
278    >>> x = np.array([-2., -1., 0., 1., 2.])
279    >>> y = np.array([-4., -2., 0., 2., 4.])
280    >>> F = C.getCostFunctionSlow(x, y)
281    >>> F([1,0,0])
282    0.0
283    >>> F([2,0,0])
284    100.0
285        """
286        #XXX: update interface to allow metric?
287        def _(params):
288            v = self.getVectorCostFunction(evalpts, observations)
289            x = v(params)
290            return sum(real((conjugate(x)*x)))
291        return _
