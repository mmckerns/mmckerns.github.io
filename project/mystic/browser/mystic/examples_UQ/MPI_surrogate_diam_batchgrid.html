36def costFactory(i):
37  """a cost factory for the cost function"""
38
39  def cost(rv):
40    """compute the diameter as a calculation of cost
41
42  Input:
43    - rv -- 1-d array of model parameters
44
45  Output:
46    - diameter -- scale * | F(x) - F(x')|**2
47    """
48    from surrogate import marc_surr as model
49
50    # prepare x and xprime
51    rv = list(rv)
52    params = rv[:-1]                         #XXX: assumes Xi' is at rv[-1]
53    params_prime = rv[:i]+rv[-1:]+rv[i+1:-1] #XXX: assumes Xi' is at rv[-1]
54
55    # get the F(x) response
56    Fx = model(params)
57
58    # get the F(x') response
59    Fxp = model(params_prime)
60
61    # compute diameter
62    scale = 1.0
63    return -scale * (Fx - Fxp)**2
64
65  return cost

71def local_optimize(cost,x0,lb,ub):
72  from mystic.solvers import PowellDirectionalSolver
73  from mystic.termination import NormalizedChangeOverGeneration as NCOG
74  from mystic.monitors import VerboseMonitor, Monitor
75
76  maxiter = 1000
77  maxfun = 1e+6
78  convergence_tol = 1e-4
79
80 #stepmon = VerboseMonitor(100)
81  stepmon = Monitor()
82  evalmon = Monitor()
83
84  ndim = len(lb)
85
86  solver = PowellDirectionalSolver(ndim)
87  solver.SetInitialPoints(x0)
88  solver.SetStrictRanges(min=lb,max=ub)
89  solver.SetEvaluationLimits(maxiter,maxfun)
90  solver.SetEvaluationMonitor(evalmon)
91  solver.SetGenerationMonitor(stepmon)
92
93  tol = convergence_tol
94  solver.Solve(cost, termination=NCOG(tol))
95
96  solved_params = solver.bestSolution
97  solved_energy = solver.bestEnergy
98  func_evals = solver.evaluations
99  return solved_params, solved_energy, func_evals

105def optimize(cost,lower,upper):
106  from mystic.tools import random_seed
107  from pyina.launchers import Mpi as Pool
108  random_seed(123)
109
110  # generate arrays of points defining a grid in parameter space
111  grid_dimensions = len(lower)
112  bins = []
113  for i in range(grid_dimensions):
114    step = abs(upper[i] - lower[i])/nbins[i]
115    bins.append( [lower[i] + (j+0.5)*step for j in range(nbins[i])] )
116
117  # build a grid of starting points
118  initial_values = gridpts(bins)
119
120  # run optimizer for each grid point
121  lb = [lower for i in range(len(initial_values))]
122  ub = [upper for i in range(len(initial_values))]
123  cf = [cost for i in range(len(initial_values))]
124  # map:: params, energy, func_evals = local_optimize(cost,x0,lb,ub)
125  results = Pool(nnodes).map(local_optimize, cf, initial_values, lb, ub)
126  #print "results = %s" % results
127
128  # get the results with the lowest energy
129  best = list(results[0][0]), results[0][1]
130  func_evals = results[0][2]
131  for result in results[1:]:
132    func_evals += result[2] # add function evaluations
133    if result[1] < best[1]: # compare energy
134      best = list(result[0]), result[1]
135
136  # return best
137  print "solved: %s" % best[0]
138  scale = 1.0
