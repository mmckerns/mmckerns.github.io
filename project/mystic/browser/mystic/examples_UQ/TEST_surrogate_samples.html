23def sampled_mean(pts,lb,ub):
24  from numpy import inf
25  f = wrap_bounds(model,lb,ub)
26  ave = 0; count = 0
27  for i in range(len(pts[0])):
28    Fx = f([pts[0][i],pts[1][i],pts[2][i]])
29    if Fx != -inf: # outside of bounds evaluates to -inf
30      ave += Fx
31      count += 1
32  if not count: return None  #XXX: define 0/0 = None
33  ave = float(ave) / float(count)

64  solver.SetStrictRanges(min=lb,max=ub)

67  solver.SetGenerationMonitor(stepmon)
68
69  tol = convergence_tol
70  solver.Solve(cost,termination=CRT(tol,tol),strategy=Best1Exp, \
71               CrossProbability=crossover,ScalingFactor=percent_change)
72
73  solved = solver.bestSolution
74 #if DEBUG: print "solved: %s" % solved
75  diameter_squared = -solver.bestEnergy / scale  #XXX: scale != 0
76  func_evals = solver.evaluations
77  return solved, diameter_squared, func_evals

168def make_cut(lb,ub,RVStart,RVend,vol):
169  global STATUS #XXX: warning! global variable...
170  params, subdiams, diam, probmass = test_cuboids(lb,ub,RVstart,RVend,vol)
171  SOLVED_PARAMETERS, SUB_DIAMETERS = params, subdiams
172  TOTAL_DIAMETERS, PROBABILITY_MASS = diam, probmass
173  if DEBUG: print "\nSTATUS = %s" % STATUS
174
175  # get region with largest probability mass
176  # region = probmass.index(max(probmass))
177  # NEW_SLICES = [region,region+1]
178
179  newstatus = []
180  newslices = []
181  # find interesting regions  (optimizer returns: solved, -Energy, f_evals)
182  for i in range(len(lb)):
183    if STATUS[i]: # status in [SUCCESS, FAILURE, UNDERSAMPLED]
184      newstatus.append(STATUS[i]) # previously determined to skip this region
185    elif not optimize(maxF,lb[i],ub[i])[1]:  # if max A = 0, then 'failure'
186      newstatus.append(FAILURE) # mark as a failure region; skip
187    elif -(optimize(minF,lb[i],ub[i])[1]): # if min A > 0, then 'success'
188      newstatus.append(SUCCESS) # mark as a success region; skip
189    else:
190      newstatus.append(None) # each 'new' slice is a bisection
191      newstatus.append(None) # ... thus appends TWO indicatiors
192      newslices.append(i)
193
194  ncut = 0
195  NEW_SLICES = []
196  for i in newslices:
197    # get direction with largest subdiameter
198    direction = subdiams[i].index(max(subdiams[i]))
199
200    # adjust for ub,lb expanded by n slices
201    region = i + ncut
202    # get the midpoint
203    cutvalue = 0.5 * ( ub[region][direction] + lb[region][direction] )
204
205    # modify bounds to include cut plane
206    l = lb[:region+1]
207    l += [lb[region][:direction] + [cutvalue] + lb[region][direction+1:]]
208    lb = l + lb[region+1:]
209    u = ub[:region]
210    u += [ub[region][:direction] + [cutvalue] + ub[region][direction+1:]]
211    ub = u + ub[region:]
212
213    # bean counting...
214    NEW_SLICES.append(region)
215    NEW_SLICES.append(region+1)
216    ncut += 1
217    TOTAL_CUTS[0] += 1
218    if TOTAL_CUTS[0] >= max_cuts[0]:
219      print "\nmaximum number of cuts performed."
220      max_cuts[0] = 0
221      STATUS = newstatus[:i+1+ncut] + STATUS[i+1:] # patially use 'old' status
222      return lb,ub
223
224  STATUS = newstatus[:]
225  return lb,ub
