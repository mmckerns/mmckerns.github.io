13def __and(*funcs):
14  "factory for compounding termination conditions with *and*"
15  def term(solver, info=False): #FIXME: 'self' doesn't flatten
16    stop = {}
17    [stop.update({f : f(solver, info)}) for f in funcs]
18    _all = all(stop.values())
19    if not info: return _all
20    if info == 'self': return tuple(stop.keys()) if _all else ()
21    return "; ".join(set("; ".join(stop.values()).split("; "))) if _all else ""
22  return term
23
24def __or(*funcs):
25  "factory for compounding termination conditions with *or*"
26  def term(solver, info=False): #FIXME: 'self' doesn't flatten
27    stop = {}
28    [stop.update({f : f(solver, info)}) for f in funcs]
29    _any = any(stop.values())
30    if not info: return _any
31    [stop.pop(cond) for (cond,met) in stop.items() if not met]
32    if info == 'self': return tuple(stop.keys())
33    return "; ".join(set("; ".join(stop.values()).split("; ")))
34  return term
