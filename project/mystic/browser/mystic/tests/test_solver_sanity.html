19def trap_stdout(): #XXX: better with contextmanager?

24def release_stdout(orig):
25    "release stdout; return any trapped output as a string"
26    out = sys.stdout.getvalue()

28    sys.stdout = orig

52        from mystic.monitors import Monitor

60        solver = self.solver
61        solver.SetRandomInitialPoints(min = self.min, max = self.max)
62        if self.usebounds: solver.SetStrictRanges(self.min, self.max)
63        if self.uselimits: solver.SetEvaluationLimits(self.maxiter, self.maxfun)
64        if self.useevalmon: solver.SetEvaluationMonitor(esow)
65        if self.usestepmon: solver.SetGenerationMonitor(ssow)
66        #### run solver, but trap output
67        _stdout = trap_stdout()
68        solver.Solve(self.costfunction, self.term, **kwds)
69        out = release_stdout(_stdout)
70        ################################
71        sol = solver.Solution()
72
73        iter=1
74       #if self.uselimits and self.maxiter == 0: iter=0
75        # sanity check solver internals
76        self.assertTrue(solver.generations == len(solver._stepmon._y)-iter)
77        self.assertTrue(list(solver.bestSolution) == solver._stepmon.x[-1]) #XXX
78        self.assertTrue(solver.bestEnergy == solver._stepmon.y[-1])
79        self.assertTrue(solver.solution_history == solver._stepmon.x)
80        self.assertTrue(solver.energy_history == solver._stepmon.y)
81        if self.usestepmon:
82            self.assertTrue(ssow.x == solver._stepmon.x)
83            self.assertTrue(ssow.y == solver._stepmon.y)
84            self.assertTrue(ssow._y == solver._stepmon._y)
85        if self.useevalmon:
86            self.assertTrue(solver.evaluations == len(solver._evalmon._y))
87            self.assertTrue(esow.x == solver._evalmon.x)
88            self.assertTrue(esow.y == solver._evalmon.y)
89            self.assertTrue(esow._y == solver._evalmon._y)

91        # Fail appropriately for solver/termination mismatch

95            self.assertTrue(warn in out)
