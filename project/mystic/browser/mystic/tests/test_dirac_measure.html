26def test_calculate_methods(npts=2):
27  upper_bounds = [1.0]

30  # -------------------------------------
31  # generate initial coordinates, weights
32  # -------------------------------------
33  # get a random distribution of points
34  if disp: print "generate random points and weights"

38
39  # calculate sample range
40  R0 = R = spread(D)
41  if disp: print "range: %s" % R
42
43  # select weights randomly in [0,1], then normalize so sum(weights) = 1
44  wts = random_samples([0],[1], npts)[0]
45  weights = normalize(wts, 0.0, zsum=True)
46  if disp: print "weights (when normalized to 0.0): %s" % weights
47  assert almostEqual(sum(weights), 0.0, tol=1e-15)
48  weights = normalize(wts, 1.0)
49  assert almostEqual(sum(weights), 1.0, tol=1e-15)
50  if disp: print "weights (when normalized to 1.0): %s" % weights
51  w = norm(weights)
52  if disp: print "norm: %s" % w
53  assert almostEqual(w, sum(weights)/npts)
54
55  # calculate sample mean
56  m0 = m = mean(D,weights)
57  if disp: print "mean: %s" % m
58  if disp: print ""
59
60  # -------------------------------------
61  # modify coordinates, maintaining mean & range
62  # -------------------------------------
63  # get new random distribution
64  if disp: print "modify positions, maintaining mean and range"
65  coordinates = samplepts(lower_bounds, upper_bounds, npts)
66  D = [i[0] for i in coordinates]

132
133  # check basic behavior for set of two points
134  s = set([point(1.0,1.0), point(3.0,2.0)]) #XXX: s + [pt, pt] => broken

144  assert almostEqual(s.range, 1.0)
145  assert almostEqual(s.mass, 1.0)
146
147  # add and remove points
148  # test special cases: SUM(weights)=0, RANGE(samples)=0, SUM(samples)=0
149  s.append(point(1.0,-1.0))
150  assert s.mean == -inf
151  assert almostEqual(s.range, 1.0)
152  assert almostEqual(s.mass, 0.0)
153

155  _ave = s.mean
156  s.mean = 1.0
157  assert str(s.mean) == 'nan'
158  assert str(s.range) == 'nan'

161  assert str(s.mass) == 'nan'
162  s.mean = _ave
163  '''

196def test_collection_behavior():
197  from mystic.math import almostEqual
198  from numpy import inf
199  def f(x): return sum(x)  # a test function for expectation value
200
201  # build three sets (x,y,z)
202  sx = set([point(1.0,1.0), point(2.0,1.0), point(3.0,2.0)])
203  sy = set([point(0.0,1.0), point(3.0,2.0)])
204  sz = set([point(1.0,1.0), point(2.0,3.0)])
205  #NOTE: for marc_surr(x,y,z), we must have x > 0.0
206  sx.normalize()
207  sy.normalize()
208  sz.normalize()
209  assert sx.mass == sy.mass == sz.mass == 1.0
210
211  # build a collection
212  c = collection([sx,sy,sz])
213  xpos = c[0].positions
214  ypos = c[1].positions
215  zpos = c[2].positions
216  xwts = c[0].weights
217  ywts = c[1].weights
218  zwts = c[2].weights
219  if disp:
220    print "x_positions: %s" % xpos
221    print "y_positions: %s" % ypos
222    print "z_positions: %s" % zpos
223    print "x_weights: %s" % xwts
224    print "y_weights: %s" % ywts
225    print "z_weights: %s" % zwts
226  assert xpos == sx.positions
227  assert ypos == sy.positions
228  assert zpos == sz.positions
229  assert xwts == sx.weights
230  assert ywts == sy.weights
231  assert zwts == sz.weights
232  tol = .2
233  supp = c.support(tol)
234  positions = c.positions
235  weights = c.weights
236  assert supp == [p for (p,w) in zip(positions,weights) if w > tol]
237  if disp:
238    print "support points:\n %s" % supp
239    print "npts: %s (i.e. %s)" % (c.npts, c.pts)
240    print "weights: %s" % weights
241    print "positions: %s" % positions
242  assert c.npts == sx.npts * sy.npts * sz.npts
243  assert len(weights) == len(positions) == c.npts
244  assert sx.positions in c.pos
245  assert sy.positions in c.pos
246  assert sz.positions in c.pos
247  assert sx.weights in c.wts
248  assert sy.weights in c.wts
249  assert sz.weights in c.wts
250
251  c_exp = c.expect(f)
252  if disp:
253    print "mass: %s" % c.mass
254    print "expect: %s" % c_exp
255  assert c.mass == [sx.mass, sy.mass, sz.mass]
256  assert c_exp == expectation(f, c.positions, c.weights)
257
258 #print "center: %s" % c.center
259 #print "delta: %s" % c.delta
260
261  # change the positions in the collection
262  points = [ list(i) for i in positions[::3] ]
263  for i in range(len(points)):
264    points[i][0] = 0.5
265  positions[::3] = points
266  c.positions = positions

268  _xpos = c[0].positions
269  _ypos = c[1].positions
270  _zpos = c[2].positions
271  _cexp = c.expect(f)

278  assert _ypos == ypos
