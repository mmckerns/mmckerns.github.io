16def ballistic_limit(h,a):
17  """calculate ballistic limit
18
19  Inputs:
20    - h = thickness
21    - a = obliquity
22
23  Outputs:
24    - v_bl = velocity (ballistic limit)
25"""
26 #assumes h,a have been scaled:
27 #h = x[0] * 25.4 * 1e-3
28 #a = x[1] * pi/180.0
29  Ho = 0.5794
30  s = 1.4004
31  n = 0.4482
32  return Ho * ( h / cos(a)**n )**s

34def marc_surr(x):
35  """calculate perforation area using a tanh-based model surrogate
36
37  Inputs:
38    - x = [thickness, obliquity, speed]
39
40  Outputs:
41    - A = perforation area
42"""
43# h = thickness = [60,105]
44# a = obliquity = [0,30]
45# v = speed = [2.1,2.8]
46  h = x[0] * 25.4 * 1e-3
47  a = x[1] * pi/180.0
48  v = x[2]
49
50  K = 10.3963
51  p = 0.4757
52  u = 1.0275
53  m = 0.4682
54  Dp = 1.778
55
56  # compare to ballistic limit
57  v_bl = ballistic_limit(h,a)
58  if v < v_bl:
59    return 0.0
60
61  return K * (h/Dp)**p * (cos(a))**u * (tanh((v/v_bl)-1))**m
