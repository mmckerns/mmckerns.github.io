42class Schwefel(AbstractFunction):
43    __doc__ = \
44    """a Schwefel's function generator
45
46Schwefel's function [1,2] has alternating rows of peaks and valleys,
47with the global minimum near the edge of the bounded parameter
48space.  This funciton can be misleading for optimizers as the next
49best local minima are near the other corners of the bounded
50parameter space.  The intensity of the peaks and valleys increases
51as one moves away from the origin.
52
53The generated function f(x) is identical to function (7)
54of [1], where len(x) >= 0.
55    """ + _doc
56    def __init__(self, ndim=2): # is n-dimensional
57        AbstractFunction.__init__(self, ndim=ndim)
58        return
59
60    def function(self,coeffs):
61        """evaluates Schwefel's function for a list of coeffs
62
63f(x) = \sum_(i=0)^(N-1) -x_i * \sin(\sqrt(\abs(x_i)))
64
65Where abs(x_i) <= 500.
66
67Inspect with mystic_model_plotter using::
68    mystic.models.schwefel -b "-500:500:10, -500:500:10" -d
69
70The minimum is f(x)=-(N+1)*418.98288727243374 at x_i=420.9687465 for all i"""
71        return sum(-c * sin(sqrt(abs(c))) for c in coeffs)
72
73    minimizers = None #XXX: there are many local minima
74    pass

77class HyperEllipsoid(AbstractFunction):
78    __doc__ = \
79    """a Pohlheim's rotated hyper-ellipsoid function generator
80
81Pohlheim's rotated hyper-ellipsoid function [1] is continuous,
82convex, and unimodal. The global minimum is located at the
83center of the N-dimensional axis parallel hyper-ellipsoid.
84
85The generated function f(x) is identical to function (1b)
86of [1], where len(x) >= 0.
87    """ + _doc
88    def __init__(self, ndim=2): # is n-dimensional
89        AbstractFunction.__init__(self, ndim=ndim)
90        return
91
92    def function(self,coeffs):
93        """evaluates the rotated hyper-ellipsoid function for a list of coeffs
94
95f(x) = \sum_(i=0)^(N-1) (\sum_(j=0)^(i) x_j)^2
96
97Inspect with mystic_model_plotter using::
98    mystic.models.ellipsoid -b "-5:5:.1, -5:5:.1" -d
99
100The minimum is f(x)=0.0 at x_i=0.0 for all i"""
101        ncoeffs = range(len(coeffs))
102        return sum(sum(coeffs[0:i+1])**2 for i in ncoeffs)
103
104    minimizers = [0.]
105    pass

108class Rastrigin(AbstractFunction):
109    __doc__ = \
110    """a Rastrigin's function generator
111
112Rastrigin's function [1] is essentially De Jong's sphere with the
113addition of cosine modulation to produce several regularly distributed
114local minima. The global minimum is at the origin.
115
116The generated function f(x) is identical to function (6)
117of [1], where len(x) >= 0.
118    """ + _doc
119    def __init__(self, ndim=2): # is n-dimensional
120        AbstractFunction.__init__(self, ndim=ndim)
121        return
122
123    def function(self,coeffs):
124        """evaluates Rastrigin's function for a list of coeffs
125
126f(x) = 10 * N + \sum_(i=0)^(N-1) (x_(i)^2 - 10 * \cos(2 * \pi * x_(i)))
127
128Inspect with mystic_model_plotter using::
129    mystic.models.rastrigin -b "-5:5:.1, -5:5:.1" -d
130
131The minimum is f(x)=0.0 at x_i=0.0 for all i"""
132        return 10.*len(coeffs) + sum(c*c - 10.*cos(2*pi*c) for c in coeffs)
133
134    minimizers = None #XXX: there are many local minima
135    pass

169class Ackley(AbstractFunction):
170    __doc__ = \
171    """an Ackley's path function generator
172
173At a very coarse level, Ackley's path function [1,3] is a slightly
174parabolic plane, with a sharp cone-shaped depression at the origin.
175The global minimum is found at the origin. There are several local
176minima evenly distributed across the function surface, where the
177surface modulates similarly to cosine.
178
179The generated function f(x) is identical to function (10)
180of [1], where len(x) >= 0.
181    """ + _doc
182    def __init__(self, ndim=2): # is n-dimensional
183        AbstractFunction.__init__(self, ndim=ndim)
184        return
185
186    def function(self,coeffs):
187        """evaluates Ackley's function for a list of coeffs
188
189f(x) = f_0(x) + f_1(x)
190
191Where:
192f_0(x) = -20 * \exp(-0.2 * \sqrt(1/N * \sum_(i=0)^(N-1) x_(i)^(2)))
193and:
194f_1(x) = -\exp(1/N * \sum_(i=0)^(N-1) cos(2 * \pi * x_(i))) + 20 + \exp(1)
195
196Inspect with mystic_model_plotter using::
197    mystic.models.ackley -b "-10:10:.1, -10:10:.1" -d
198
199The minimum is f(x)=0.0 at x_i=0.0 for all i"""
200        a=20.; b=0.2; d=2*pi
201        n = len(coeffs)
202        f0 = -a * exp(-b * sqrt(sum(c*c for c in coeffs)/n))
203        f1 = -exp(sum(cos(d*c) for c in coeffs)/n) + a + exp(1)
204        return f0 + f1
205
206    minimizers = None #XXX: there are many local minima
207    pass

210class Michalewicz(AbstractFunction):
211    __doc__ = \
212    """a Michalewicz's function generator
213
214Michalewicz's function [1,4] in general evaluates to zero. However,
215there are long narrow channels that create local minima. At the
216intersection of the channels, the function additionally has sharp
217dips -- one of which is the global minimum.
218
219The generated function f(x) is identical to function (12)
220of [1], where len(x) >= 0.
221    """ + _doc
222    def __init__(self, ndim=5): # is n-dimensional
223        AbstractFunction.__init__(self, ndim=ndim)
224        return
225
226    def function(self,coeffs):
227        """evaluates Michalewicz's function for a list of coeffs
228
229f(x) = -\sum_(i=0)^(N-1) \sin(x_i) * (\sin((i+1) * (x_i)^(2) / \pi))^(20)
230
231Inspect with mystic_model_plotter using::
232    mystic.models.michal -b "0:3.14:.1, 0:3.14:.1, 1.28500168, 1.92305311, 1.72047194" -d
233
234For x=(2.20289811, 1.57078059, 1.28500168, 1.92305311, 1.72047194, ...)[:N]
235and c=(-0.801303, -1.0, -0.959092, -0.896699, -1.030564, ...)[:N],
236the minimum is f(x)=sum(c) for all x_i=(0,pi)"""
237        m = 20 # the larger the number the narrower the channel
238        n = range(len(coeffs))
239        r = -sum(sin(coeffs[i])*(sin((i+1)*coeffs[i]**2/pi))**m for i in n)
240        return r #XXX: constrain x_i=(0,pi) ?
241
242    minimizers = None #XXX: there are many local minima
243    pass

246class Branins(AbstractFunction):
247    __doc__ = \
248    """a Branins's rcos function generator
249
250Branins's function [1,5] is very similar to Rosenbrock's saddle function.
251However unlike Rosenbrock's saddle, Branins's function has a degenerate
252global minimum.
253
254The generated function f(x) is identical to function (13)
255of [1], where len(x) >= 0.
256    """ + _doc
257    def __init__(self, ndim=2): # is 2-dimensional
258        AbstractFunction.__init__(self, ndim=ndim)
259        return
260
261    def function(self,coeffs):
262        """evaluates Branins's function for a list of coeffs
263
264f(x) = f_0(x) + f_1(x)
265
266Where:
267f_0(x) = a * (x_1 - b * x_(0)^(2) + c * x_0 - d)^2
268and
269f_1(x) = e * (1 - f) * \cos(x_0) + e
270and
271a=1, b=5.1/(4*pi^2), c=5/pi, d=6, e=10, f=1/(8*pi)
272
273Inspect with mystic_model_plotter using::
274    mystic.models.branins -b "-10:20:.1, -5:25:.1" -d -x 1
275
276The minimum is f(x)=0.397887 at x=((2 +/- (2*i)+1)*pi, 2.275 + 10*i*(i+1)/2)
277for all i"""
278        x1,x2 = coeffs
279        a=1.; b=5.1/(4*pi**2); c=5./pi; d=6.; e=10.; f=1./(8*pi)
280        f0 = a * (x2 - b * x1**2 + c * x1 - d)**2
281        f1 = e * (1 - f) * cos(x1) + e
282        return f0 + f1
283
284    minimizers = None #XXX: there are degenerate global minima
285    pass

288class Easom(AbstractFunction):
289    __doc__ = \
290    """a Easom's function generator
291
292Easom's function [1,6] is a unimodal function that exaluates
293to zero everywhere except in the region around the global
294minimum. The global minimum is at the bottom of a sharp well.
295
296The generated function f(x) is identical to function (14)
297of [1], where len(x) >= 0.
298    """ + _doc
299    def __init__(self, ndim=2): # is 2-dimensional
300        AbstractFunction.__init__(self, ndim=ndim)
301        return
302
303    def function(self,coeffs):
304        """evaluates Easom's function for a list of coeffs
305
306f(x) = -\cos(x_0) * \cos(x_1) *  \exp(-((x_0-\pi)^2+(x_1-\pi)^2))
307
308Inspect with mystic_model_plotter using::
309    mystic.models.easom -b "-5:10:.1, -5:10:.1" -d
310
311The minimum is f(x)=-1.0 at x=(pi,pi)"""
312        x0,x1 = coeffs
313        return -cos(x0) * cos(x1) *  exp(-((x0-pi)**2+(x1-pi)**2))
314
315    minimizers = [(pi,pi)]
316    pass

319class GoldsteinPrice(AbstractFunction):
320    __doc__ = \
321    """a Goldstein-Price's function generator
322
323Goldstein-Price's function [1,7] provides a function with several
324peaks surrounding a roughly flat valley. There are a few shallow
325scorings across the valley, where the global minimum is found at
326the intersection of the deepest of the two scorings. Local minima
327occur at other intersections of scorings.
328
329The generated function f(x) is identical to function (15)
330of [1], where len(x) >= 0.
331    """ + _doc
332    def __init__(self, ndim=2): # is 2-dimensional
333        AbstractFunction.__init__(self, ndim=ndim)
334        return
335
336    def function(self,coeffs):
337        """evaluates Goldstein-Price's function for a list of coeffs
338
339f(x) = (1 + (x_0 + x_1 + 1)^2 * f_0(x)) * (30 + (2*x_0 - 3*x_1)^2 * f_1(x))
340
341Where:
342f_0(x) = 19 - 14*x_0 + 3*x_(0)^2 - 14*x_1 + 6*x_(0)*x_(1) + 3*x_(1)^2
343and
344f_1(x) = 18 - 32*x_0 + 12*x_(0)^2 + 48*x_1 - 36*x_(0)*x_(1) + 27*x_(1)^2
345
346Inspect with mystic_model_plotter using::
347    mystic.models.goldstein -b "-5:5:.1, -5:5:.1" -d -x 1
348
349The minimum is f(x)=3.0 at x=(0,-1)"""
350        x0,x1 = coeffs
351        f0 = 19 - 14*x0 + 3*x0**2 - 14*x1 + 6*x0*x1 + 3*x1**2
352        f1 = 18 - 32*x0 + 12*x0**2 + 48*x1 - 36*x0*x1 + 27*x1**2
353        return (1 + (x0 + x1 + 1)**2 * f0) * (30 + (2*x0 - 3*x1)**2 * f1)
354
355    minimizers = [(0,-1),(-.6,-.4),(1.8,.2)]
356    pass
