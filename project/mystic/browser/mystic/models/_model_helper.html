15def chebyshev(trial, target, M=61):
16    from mystic.math import polyeval
17    result=0.0
18    x=-1.0
19    dx = 2.0 / (M-1)
20    for i in range(M):
21        px = polyeval(trial, x)
22        if px<-1 or px>1:
23            result += (1 - px) * (1 - px)
24        x += dx
25
26    px = polyeval(trial, 1.2) - polyeval(target, 1.2)
27    if px<0: result += px*px
28
29    px = polyeval(trial, -1.2) - polyeval(target, -1.2)
30    if px<0: result += px*px
31
32    return result
