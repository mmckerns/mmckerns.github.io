23class BevingtonDecay(AbstractModel):
24    """
25Computes dual exponential decay [5].
26y = a1 + a2 Exp[-t / a4] + a3 Exp[-t/a5]
27    """
28
29    def __init__(self,name='decay',metric=lambda x: numpysum(x*x)):
30        AbstractModel.__init__(self,name,metric)
31        return

33    def evaluate(self,coeffs,evalpts):
34        """evaluate dual exponential decay with given coeffs over given evalpts
35coeffs = (a1,a2,a3,a4,a5)"""
36        a1,a2,a3,a4,a5 = coeffs
37        t = asarray(evalpts) #XXX: requires a numpy.array
38        return a1 + a2*exp(-t/a4) + a3*exp(-t/a5)
39
40    def ForwardFactory(self,coeffs):
41        """generates a dual decay model instance from a list of coefficients"""
42        a1,a2,a3,a4,a5 = coeffs
43        def forward_decay(evalpts):
44            """a dual exponential decay over a 1D numpy array
45with (a1,a2,a3,a4,a5) = (%s,%s,%s,%s,%s)""" % (a1,a2,a3,a4,a5)
46            return self.evaluate((a1,a2,a3,a4,a5),evalpts)
47        return forward_decay

49    def CostFactory(self,target,pts):
50        """generates a cost function instance from list of coefficients & evaluation points"""
51        datapts = self.evaluate(target,pts)
52        F = CF()
53        F.addModel(self.ForwardFactory,len(target),self.__name__)
54        self.__cost__ = F.getCostFunction(evalpts=pts,observations=datapts,sigma=sqrt(datapts),metric=self.__metric__)
55        return self.__cost__
