60def de_solve(CF, a4=None, a5=None):
61    """solve with DE for given cost funciton; fix a4 and/or a5 if provided"""
62    minrange = [0, 100, 100, 1, 1]
63    maxrange = [100, 2000, 200, 200, 200]
64    interval = 10
65    if a5 != None:
66        minrange[4] = maxrange[4] = a5
67        interval = 20
68    if a4 != None:
69        minrange[3] = maxrange[3] = a4
70        if interval == 20: interval = 1000
71
72    solver = DifferentialEvolutionSolver(ND, NP)
73    solver.enable_signal_handler()
74    stepmon = MyMonitor(interval)
75    solver.SetRandomInitialPoints(min=minrange,max=maxrange)
76    solver.SetStrictRanges(min=minrange, max=maxrange)
77    solver.SetEvaluationLimits(generations=MAX_GENERATIONS)
78    solver.SetGenerationMonitor(stepmon)
79    solver.Solve(CF,termination=ChangeOverGeneration(generations=50))
80    solution = solver.Solution()
