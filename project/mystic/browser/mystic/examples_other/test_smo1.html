69def getIndexSets(alpha, a, b, y):
70    """See Kerthi and Gilbert, and MathSVM.m"""
71    # it is either one loop, or five smart ops.
72    # will do former so it can be translated into C
73    I0,I1,I2,I3,I4 = [],[],[],[],[]
74    for i, ali, ai, bi, yi in zip(range(y.size), alpha, a, b, y):
75        if ali > ai and ali < bi:
76          I0.append(i)
77        elif ali == ai:
78          if yi > 0:
79             I1.append(i)
80          else:
81             I4.append(i)
82        else:
83          if yi < 0:
84             I2.append(i)
85          else:
86             I3.append(i)

95def QPOptimalQ(Iub, Ilb, F, tau):
96    return max(F[Ilb]) - min(F[Iub]) <= tau

98def ViolatingPairQ(Sub, Slb, i, j, F, tau):
99    # Sub and Slb should now be sets
100    if Sub.__contains__(i) and Slb.__contains__(j) and F[j] - F[i] > tau:
101        return True
102    elif Slb.__contains__(i) and Sub.__contains__(j) and F[i] - F[j] > tau:
103        return True
104    else:
105        return False

107def getViolatingPair(Iint, Iub, Ilb, F, tau):
108    # The heuristic is to focus attention to Iint
109    FIint = F[Iint]
110    if FIint and max(FIint) - min(FIint) > tau:
111         # the first index set (is nonempty and) has offending elements.
112         ihigh,ilow = FIint.argmax(), FIint.argmin()
113         return Iint[ilow], Iint[ihigh]
114    else:
115         # nothing within the first index set, need to scan
116         # MathSVM is dumber than usual
117         Sub,Slb = set(Iub), set(Ilb)
118         l = F.size
119         for i in range(l):
120             for j in range(i,l):
121                 if ViolatingPairQ(Sub, Slb, i, j, F, tau):
122                      return i,j
123    # the fact that this function is called means that it
124    # should never reach here
125    return None

128def smo_sub2d(Q, y, alpha, X, i, j, ym, yp):
129    # 2d subproblem for SMO, varying only indices i and j
130    a1,a2 = alpha[i], alpha[j]
131    y1,y2 = y[i], y[j]
132    wv = WeightVector(alpha, X, y)
133    ii = inner(wv, X)
134    bias = -0.5 * (max(ii[ym]) + min(ii[yp]))
135    ay = transpose(alpha * y)
136    TT = dot(Q, transpose(alpha * y)) + b
137    E1, E2 = TT[i] - y1, TT[j] - y2
138    k = Q[i,i] + Q[j,j] - 2. * Q[i,j]
139    C = b[0] # note, i am now assuming that a=0, b=c
140    if y1 != y2:
141        U = max(0, a2-a1)
142        V = min(C, C-a1 + a2)
143        print "1 U/V: ", U, V
144    else:
145        U = max(0., a1+a2-C)
146        V = min(C, a1+a2)
147        print "2 U/V: ", U, V
148    a2trial = a2 + y2*(E1-E2)/k
149    if a2trial > V:
150        a2new = V
151    elif a2trial < U:
152        a2new = U
153    else:
154        a2new = a2trial
155    a1new = a1 + y1*y2*(a2-a2new)
156    print "E1/E2:", E1,E2
157    print "C:",C
158    return a1new, a2new

160def QP_smo(Q, p, a, b, c, y, tau, a0, X):
161    """\
162Minimizes xQx + Px,
163   subject to a_i <= x_i <= b_i for all i
164   and y.x = c
165
166   X are the array of input points (their labels should be in y)
167    """
168    # initial alpha (must be feasible)
169    alpha = a0 
170    l = alpha.size
171    ym,yp = (y<0).nonzero()[0], (y>0).nonzero()[0]
172    while 1:
173        Isets = getIndexSets(alpha, a, b, y)
174        Iub = getIub(*Isets)
175        Ilb = getIlb(*Isets)
176        F = (dot(Q, alpha) + p)/y
177        if QPOptimalQ(Iub, Ilb, F, tau):
178            break
179        B = getViolatingPair(Isets[0], Iub, Ilb, F, tau)
180        # now solve the 2d subproblem
181        aa = smo_sub2d(Q, y, alpha, X, B[0], B[1], ym,yp)
182        print "aa: ", aa
183        alpha[B[0]], alpha[B[1]] = aa
184        print alpha
185        break
