8"""
9Given a set of points in the plane, find the smallest circle
10that contains them. (using DE and scipy.fmin)
11
12Requires:
13  -- numpy, pylab
14
15The pylab output will draw
16  -- a set of points inside a circle defined by x0,y0,R0
17  -- the circle (x0,y0) with rad R0
18  -- the optimized circle with minimum R enclosing the points
19"""
20
21from mystic.models import circle, sparse_circle
22import pylab
23
24# generate training set & define cost function
25# CostFactory2 allows costfunction to reuse datapoints from training set
26x0, y0, R0 = [10., 20., 3]
27npts = 20
28xy = sparse_circle(x0, y0, R0, npts)
29cost = sparse_circle.CostFactory2(xy)
30
31# function to find the 'support vectors' given R
32# SV in quotes because they are found by optimizing the primal,
33# as opposed to "directly" via the dual.
34def sv(data, xx,yy,rr):
35    svl = []
36    for i in range(len(data)):
37       x,y = data[i]
38       if abs((xx-x)*(xx-x)+(yy-y)*(yy-y) - rr*rr) < 0.01:
39           svl.append(i)
40    return svl
41
42# DEsolver inputs
43MAX_GENERATIONS = 2000
44ND, NP = 3, 30    # dimension, population size
45minrange = [0., 0., 0.]
46maxrange = [50., 50., 10.]
47
48# prepare DESolver
49from mystic.solvers import DifferentialEvolutionSolver2 \
50      as DifferentialEvolutionSolver
51from mystic.termination import ChangeOverGeneration, VTR
52solver = DifferentialEvolutionSolver(ND, NP)
53solver.enable_signal_handler()
54solver.SetRandomInitialPoints(min=minrange,max=maxrange)
55solver.SetEvaluationLimits(generations=MAX_GENERATIONS)
56solver.Solve(cost, termination=ChangeOverGeneration(generations=100))
57
