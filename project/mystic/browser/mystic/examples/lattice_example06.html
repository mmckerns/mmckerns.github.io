7"""
8Example:
9    - Solve 8th-order Chebyshev polynomial coefficients with Powell's method.
10    - Uses LatticeSolver to provide 'pseudo-global' optimization
11    - Plot of fitting to Chebyshev polynomial.
12
13Demonstrates:
14    - standard models
15    - minimal solver interface
16"""
17# the Lattice solver
18from mystic.solvers import LatticeSolver
19
20# Powell's Directonal solver
21from mystic.solvers import PowellDirectionalSolver
22
23# Chebyshev polynomial and cost function
24from mystic.models.poly import chebyshev8, chebyshev8cost
25from mystic.models.poly import chebyshev8coeffs
26
27# if available, use a pathos worker pool
28try:
29    from pathos.pools import ProcessPool as Pool
30   #from pathos.pools import ParallelPool as Pool
31except ImportError:
32    from mystic.pools import SerialPool as Pool
33
34# tools
35from mystic.termination import NormalizedChangeOverGeneration as NCOG
36from mystic.math import poly1d
37from mystic.monitors import VerboseMonitor
38from mystic.tools import getch
39import pylab
40pylab.ion()
41
42# draw the plot
43def plot_exact():
44    pylab.title("fitting 8th-order Chebyshev polynomial coefficients")
45    pylab.xlabel("x")
46    pylab.ylabel("f(x)")
47    import numpy
48    x = numpy.arange(-1.2, 1.2001, 0.01)
49    exact = chebyshev8(x)
50    pylab.plot(x,exact,'b-')
51    pylab.legend(["Exact"])
52    pylab.axis([-1.4,1.4,-2,8],'k-')
53    pylab.draw()
54    return

56# plot the polynomial
57def plot_solution(params,style='y-'):
58    import numpy
59    x = numpy.arange(-1.2, 1.2001, 0.01)
60    f = poly1d(params)
61    y = f(x)
62    pylab.plot(x,y,style)
63    pylab.legend(["Exact","Fitted"])
64    pylab.axis([-1.4,1.4,-2,8],'k-')
65    pylab.draw()
66    return

69if __name__ == '__main__':
70    from pathos.helpers import freeze_support
71    freeze_support() # help Windows use multiprocessing
72
73    print "Powell's Method"
74    print "==============="
75
76    # dimensional information
77    from mystic.tools import random_seed
78    random_seed(123)
79    ndim = 9
80    nbins = 8 #[2,1,2,1,2,1,2,1,1]
81
82    # draw frame and exact coefficients
83    plot_exact()
84
85    # configure monitor
86    stepmon = VerboseMonitor(1)
87
88    # use lattice-Powell to solve 8th-order Chebyshev coefficients
89    solver = LatticeSolver(ndim, nbins)
90    solver.SetNestedSolver(PowellDirectionalSolver)
91    solver.SetMapper(Pool().map)
92    solver.SetGenerationMonitor(stepmon)
93    solver.SetStrictRanges(min=[-300]*ndim, max=[300]*ndim)
94    solver.Solve(chebyshev8cost, NCOG(1e-4), disp=1)
95    solution = solver.Solution()
96
97    # use pretty print for polynomials
98    print poly1d(solution)
99
100    # compare solution with actual 8th-order Chebyshev coefficients
101    print "\nActual Coefficients:\n %s\n" % poly1d(chebyshev8coeffs)
102
103    # plot solution versus exact coefficients
104    plot_solution(solution)
105    getch()
