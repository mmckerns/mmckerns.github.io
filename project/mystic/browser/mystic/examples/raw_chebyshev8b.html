8""" NOTE:

13from mystic.models.poly import chebyshev8coeffs as target
14from mystic.math import polyeval, poly1d
15
16def chebyshev8cost(trial,M=61):
17    """The costfunction for order-n Chebyshev fitting.
18M evaluation points between [-1, 1], and two end points"""
19
20    result=0.0
21    x=-1.0
22    dx = 2.0 / (M-1)
23    for i in range(M):
24        px = polyeval(trial, x)
25        if px<-1 or px>1:
26            result += (1 - px) * (1 - px)
27        x += dx
28
29    px = polyeval(trial, 1.2) - polyeval(target, 1.2)
30    if px<0: result += px*px
31
32    px = polyeval(trial, -1.2) - polyeval(target, -1.2)
33    if px<0: result += px*px
34
35    return result
36
