29def ForwardMogiFactory(params):
30    x0,y0,z0,dV = params
31    def forward_mogi(evalpts):
32        """ evalpts should be a 2D (2 by N) numpy array """
33        dx = evalpts[0,:] - x0
34        dy = evalpts[1,:] - y0
35        dz = 0 - z0
36        c = dV * 3. / 4. * pi
37        # or equivalently c= (3/4) a^3 dP / rigidity
38        # where a = sphere radius, dP = delta Pressure
39        r2 = dx*dx + dy*dy + dz*dz
40        C = c / pow(r2, 1.5)
41        return array((C*dx,C*dy,C*dz))
42    return forward_mogi

120class MogiModel(Model):
121    """a park model:
122 - parameters are passed as named strings to set them as class attributes
123 - function that does the evaluation must be named "eval"
124 - __call__ generated that takes namestring and parameter-named keywords
125"""
126    parameters = ["x0","y0","z0","dV"]
127    def eval(self, x):
128        x0 = self.x0
129        y0 = self.y0
130        z0 = self.z0
131        dV = self.dV
132        f = ForwardMogiFactory((x0,y0,z0,dV))
133        return f(x)
134    pass
