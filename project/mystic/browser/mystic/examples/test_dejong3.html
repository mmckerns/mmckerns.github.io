23from mystic.solvers import DifferentialEvolutionSolver
24from mystic.termination import ChangeOverGeneration, VTR
25from mystic.models.dejong import step as DeJong3
26
27from mystic.tools import random_seed
28random_seed(123)
29
30ND = 5
31NP = 25
32MAX_GENERATIONS = 2500
33
34def main():
35    solver = DifferentialEvolutionSolver(ND, NP)
36
37    solver.SetRandomInitialPoints(min = [-5.12]*ND, max = [5.12]*ND)
38    solver.SetEvaluationLimits(generations=MAX_GENERATIONS)
39
40    solver.Solve(DeJong3, termination=VTR(0.00001), \
41                 CrossProbability=0.3, ScalingFactor=1.0)
42
43    solution = solver.Solution()
44 
45    print solution
46
47

55    print "CPU Time: %s\n" % timetaken
