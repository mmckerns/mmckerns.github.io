8"""
9The Twisted Gausssian in
10"Shuffled Complex Evolution Metropolis" Algoritm of Vrugt et al. [1]
11
12Reference:
13
14[1] Jasper A. Vrugt, Hoshin V. Gupta, Willem Bouten, and Soroosh Sorooshian
15A Shuffled Complex Evolution Metropolis algorithm for optimization and uncertainty assessment of hydrologic model parameters
16WATER RESOURCES RESEARCH, VOL. 39, NO. 8, 1201, doi:10.1029/2002WR001642, 2003
17Link to paper: http://www.agu.org/pubs/crossref/2003/2002WR001642.shtml
18
19[2] Vrugt JA, Nuallain , Robinson BA, Bouten W, Dekker SC, Sloot PM
20Application of parallel computing to stochastic parameter estimation in environmental models
21Computers & Geosciences, Vol. 32, No. 8. (October 2006), pp. 1139-1155.
22Link to paper: http://www.science.uva.nl/research/scs/papers/archive/Vrugt2006b.pdf
23
24"""
25
26from mystic.scemtools import *
27from numpy import zeros, identity, array
28from numpy import random
29
30# dimension of density function
31n = 2
32
33# number of parallel chains
34q = 10
35
36# number of points per complex
37m = 100
38
39m1 = zeros(n)
40S1 = identity(n)
41S1[0,0] = 100
42
43def twist(X):
44    b = 0.1
45    Y = array(X)*1.
46    Y[1] += b * X[0]**2 - 100. * b
47    return Y
48
49p = multinormal_pdf(m1,S1)
50def target(X):
51    return p(twist(X))     
52
53def proposal(X):
54    return random.multivariate_normal(X, 10. * identity(n))
55
56def initpop(npts, ndim):
57    return random.rand(npts, ndim) * 200. -1
58
59a = initpop(q*m, n)
60Cs = sort_and_deal(a, target, q)
61Ck = Cs[0] # 0 for the first deal, -1 for the last
62ak = [target(c) for c in Ck]
63Sk = [ Ck[0] ]
64Sak = [ ak[0] ]
65L = 10000
66
67if __name__=='__main__':
68    from mystic.metropolis import *
69    import time
70
71    t1 = time.time()
72    for i in xrange(L):
73        scem(Ck, ak, Sk, Sak, target, 0.1)
74    t2 = time.time()
75    print "SCEM 1 chain for x[%d] took %0.3f ms" % (len(Sk), (t2-t1)*1000)
76    Sk = array(Sk)
77
78    t1 = time.time()
79    x = [ [0,10] ]
80    for i in xrange(L):
81        x.append(metropolis_hastings(proposal, target, x[-1]))
82    t2 = time.time()
83    print "2D Metropolis for x[%d] took %0.3f ms" % (len(x), (t2-t1)*1000)
84    x = array(x)
85
86    #import dill
87    #dill.dump(x, open('twisted1.pkl','w'))
88    #dill.dump(Sk, open('twisted1.pkl','w'))
89
90    import pylab
91    pylab.plot(Sk[:,0],Sk[:,1],'r.')
92    pylab.plot(x[:,0] + 30,x[:,1],'b.')
93    pylab.show()
