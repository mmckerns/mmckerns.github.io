23def plot_solution(func):
24    try:
25        import Gnuplot, numpy
26        g = Gnuplot.Gnuplot(debug = 1)
27        x = numpy.arange(-1.2, 1.2001, 0.01)
28        x2 = numpy.array([-1.0, 1.0])
29        p = poly1d(func)
30        chebyshev = poly1d(Chebyshev8)
31        y = p(x)
32        exact = chebyshev(x)
33        # g('set style line lw 5')
34        g('set xrange[-1.4:1.4]')
35        g('set yrange[-2:8]')
36        g.plot(Gnuplot.Data(x, y, with_='line', inline=1, title="Storn and Price's Function Fitting problem"), \
37               Gnuplot.Data(x, exact, with_='l 4 4'), \
38               Gnuplot.Data(x, 0*x-1, with_='l 2 2'), \
39               Gnuplot.Data(x2, 0*x2+1, with_='l 2 2'),  \
40               Gnuplot.Data([-1.2, -1.2], [-1, 10], with_='l 2 2'),  \
41               Gnuplot.Data([1.2, 1.2], [-1, 10], with_='l 2 2'),  \
42               Gnuplot.Data([-1.0, -1.0], [1, 10], with_='l 2 2'),  \
43               Gnuplot.Data([1.0, 1.0], [1, 10], with_='l 2 2')  \
44               )
45        getch('Press any key to exit plot')
46    except ImportError:
47        print "Install Gnuplot and Gnuplot.py for visualization"
48        pass
49
50
51if __name__ == '__main__':
52    solution = main()
53    print_solution(solution)
54    plot_solution(solution)
