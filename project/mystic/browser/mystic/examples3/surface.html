11class Surface(object):
12   #surface has:
13   #    args - interpolation configuration (smooth, function, ...)
14   #    sampler - a search algorithm
15   #    maxpts - a maximum number of sampling points
16   #    noise - a noise coefficient
17   #    dim - dimensionality of the model
18   #    function - target function [F(*x)]
19   #    objective - target function [F(x)]
20   #    surrogate - interpolated function [F(*x)]
21   #    model - interpolated function [F(x)]
22   #    x,y,z - sampled points(*)
23   #
24   #surface (or sampler) has:
25   #    _minmon,_maxmon - step monitor(*)
26   #    _minarch,_maxarch - sampled point archives(*)
27   #
28   #surface can:
29   #    Sample - populate sampled point archive with solver trajectories
30   #    Interpolate - build interpolated function from sampled points
31   #    Plot - plot sampled points and interpolated surface
32   #
33   #surface (or sampler) can:
34   #    UseMonitor - track trajectories with a monitor(s)
35   #    UseArchive - track sampled points in an archive(s)
36   #    _noise - remove duplicate sampled points (x) by adding noise to x
37   #    _downsample - skip sampled points at a regular interval (for speed)
38   #    _max - fetch (x,y,z,model(x,y)) for maximal z of sampled points
39   #    _min - fetch (x,y,z,model(x,y)) for minimal z of sampled points

101    def Sample(self, bounds, stop, clear=False, verbose=False):
102        #XXX: does the strategy of finding min/max always apply?
103        import numpy as np
104
105        # get model (for minima)
106        model = self.objective
107        self.dim = len(bounds)
108
109        ### get mins ###
110        stepmon = self._minmon
111        archive = None if clear else self._minarch
112        inverse = False
113
114        self.sampler.Reset(archive, inv=inverse) # reset the sampler
115        self.sampler.Search(model, bounds, stop=stop, monitor=stepmon)
116        if verbose: self.sampler._summarize()
117        # read trajectories from log (or monitor)
118        xyz = self.sampler.Samples()
119        if clear: self.sampler.Reset()  # reset the sampler
120        ### end mins ###
121
122        # invert model (for maxima)
123        imodel = lambda *args, **kwds: -model(*args, **kwds)
124
125        ### get maxs ###
126        stepmon = self._maxmon
127        archive = None if clear else self._maxarch
128        inverse = True
129
130        self.sampler.Reset(archive, inv=inverse) # reset the sampler
131        self.sampler.Search(imodel, bounds, stop=stop, monitor=stepmon)
132        if verbose: self.sampler._summarize()
133        xyz = np.hstack((xyz, self.sampler.Samples()))
134        if clear: self.sampler.Reset()  # reset the sampler
135        ### end maxs ###
136
137        # split into params and cost
138        self.x = xyz.T[:,:-1]
139        self.z = xyz.T[:,-1]
140        return self.x, self.z

153        if maxpts is None: maxpts = self.maxpts
154        if x is None: x = self.x

164        #   plt.show()
